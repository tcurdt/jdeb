016package org.vafer.jdeb.descriptors;
017
018import java.io.BufferedReader;
019import java.io.IOException;
020import java.io.InputStream;
021import java.io.InputStreamReader;
022import java.io.StringReader;
023import java.text.ParseException;
024import java.util.HashSet;
025import java.util.LinkedHashMap;
026import java.util.Map;
027import java.util.Set;
028
029import org.vafer.jdeb.utils.Utils;
030import org.vafer.jdeb.utils.VariableResolver;
031
032/**
033 * A descriptor holds the usual key value pairs.
034 *
035 * @author Torsten Curdt
036 * @see <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html">Debian Policy Manual - Control files and their fields</a>
037 */
038public abstract class AbstractDescriptor {
039
040    protected final Map<String, String> values = new LinkedHashMap<String, String>();
041    protected final VariableResolver resolver;
042
043    private static String openToken = "[[";
044    private static String closeToken = "]]";
045
046    public AbstractDescriptor( final VariableResolver pResolver ) {
047        resolver = pResolver;
048    }
049
050    public AbstractDescriptor( final AbstractDescriptor pDescriptor ) {
051        this(pDescriptor.resolver);
052        values.putAll(pDescriptor.values);
053    }
054
055    public static void setOpenToken( final String pToken ) {
056        openToken = pToken;
057    }
058
059    public static void setCloseToken( final String pToken ) {
060        closeToken = pToken;
061    }
062
063    protected void parse( final InputStream pInput ) throws IOException, ParseException {
064        final BufferedReader br = new BufferedReader(new InputStreamReader(pInput));
065        StringBuilder buffer = new StringBuilder();
066        String key = null;
067        int linenr = 0;
068        while (true) {
069            final String line = br.readLine();
070
071            if (line == null) {
072                if (buffer.length() > 0) {
073                    // flush value of previous key
074                    set(key, buffer.toString());
075                    buffer = null;
076                }
077                break;
078            }
079
080            linenr++;
081
082            if (line.length() == 0) {
083                throw new ParseException("Empty line", linenr);
084            }
085
086            final char first = line.charAt(0);
087            if (Character.isLetter(first)) {
088
089                // new key
090
091                if (buffer.length() > 0) {
092                    // flush value of previous key
093                    set(key, buffer.toString());
094                    buffer = new StringBuilder();
095                }
096
097
098                final int i = line.indexOf(':');
099
100                if (i < 0) {
101                    throw new ParseException("Line misses ':' delimiter", linenr);
102                }
103
104                key = line.substring(0, i);
105                buffer.append(line.substring(i + 1).trim());
106
107                continue;
108            }
109
110            // continuing old value
111            buffer.append('\n').append(line.substring(1));
112        }
113        br.close();
114
115    }
116
117    public void set(final String pKey, final String pValue) {
118        String value = Utils.replaceVariables(resolver, pValue, openToken, closeToken);
119        if ("".equals(value)) {
120            value = null;
121        }
122        values.put(pKey, value);
123    }
124
125    public String get( String key ) {
126        return values.get(key);
127    }
128
129    public abstract String[] getMandatoryKeys();
130
131    public boolean isValid() {
132        return invalidKeys().size() == 0;
133    }
134
135    public Set<String> invalidKeys() {
136        final Set<String> invalid = new HashSet<String>();
137        
138        for (String aMk : getMandatoryKeys()) {
139            if (get(aMk) == null) {
140                invalid.add(aMk);
141            }
142        }
143
144        return invalid;
145    }
146
147    public String toString( final String[] keys ) {
148        StringBuilder s = new StringBuilder();
149        for (String key : keys) {
150            String value = values.get(key);
151            if (value != null) {
152                s.append(key).append(":");
153
154                try {
155                    BufferedReader reader = new BufferedReader(new StringReader(value));
156                    String line;
157                    while ((line = reader.readLine()) != null) {
158                        if (line.length() != 0 && !Character.isWhitespace(line.charAt(0))) {
159                            s.append(' ');
160                        }
161
162                        s.append(line).append('\n');
163                    }
164                } catch (IOException e) {
165                    e.printStackTrace();
166                }
167            }
168        }
169        return s.toString();
170    }
171}
