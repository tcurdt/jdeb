016package org.vafer.jdeb.maven;
017
018import static org.vafer.jdeb.maven.MissingSourceBehavior.FAIL;
019import static org.vafer.jdeb.maven.MissingSourceBehavior.IGNORE;
020
021import java.io.File;
022import java.io.FileNotFoundException;
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.List;
026import java.util.StringTokenizer;
027
028import org.vafer.jdeb.DataConsumer;
029import org.vafer.jdeb.DataProducer;
030import org.vafer.jdeb.producers.DataProducerArchive;
031import org.vafer.jdeb.producers.DataProducerDirectory;
032import org.vafer.jdeb.producers.DataProducerFile;
033import org.vafer.jdeb.producers.DataProducerLink;
034import org.vafer.jdeb.producers.DataProducerPathTemplate;
035
036/**
037 * Maven "data" element acting as a factory for DataProducers. So far Archive and
038 * Directory producers are supported. Both support the usual ant pattern set
039 * matching.
040 *
041 * @author Bryan Sant
042 */
043public final class Data implements DataProducer {
044
045    private File src;
046
047    /**
048     * @parameter expression="${src}"
049     */
050    public void setSrc( File src ) {
051        this.src = src;
052    }
053
054    private String dst;
055
056    /**
057     * @parameter expression="${dst}"
058     */
059    public void setDst( String dst ) {
060        this.dst = dst;
061    }
062
063    private String type;
064
065    /**
066     * @parameter expression="${type}"
067     */
068    public void setType( String type ) {
069        this.type = type;
070    }
071
072    private MissingSourceBehavior missingSrc = FAIL;
073
074    /**
075     * @parameter expression="${missingSrc}"
076     */
077    public void setMissingSrc( String missingSrc ) {
078        MissingSourceBehavior value = MissingSourceBehavior.valueOf(missingSrc.trim().toUpperCase());
079        if (value == null) {
080            throw new IllegalArgumentException("Unknown " + MissingSourceBehavior.class.getSimpleName() + ": " + missingSrc);
081        }
082        this.missingSrc = value;
083    }
084
085    private String linkName;
086
087    /**
088     * @parameter expression="${linkName}"
089     */
090    public void setLinkName(String linkName) {
091        this.linkName = linkName;
092    }
093
094    private String linkTarget;
095
096    /**
097     * @parameter expression="${linkTarget}"
098     */
099    public void setLinkTarget(String linkTarget) {
100        this.linkTarget = linkTarget;
101    }
102
103    private boolean symlink;
104
105    /**
106     * @parameter expression="${symlink}"
107     */
108    public void setSymlink(boolean symlink) {
109        this.symlink = symlink;
110    }
111
112    private String[] includePatterns;
113
114    /**
115     * @parameter expression="${includes}" alias="includes"
116     */
117    public void setIncludes( String includes ) {
118        includePatterns = splitPatterns(includes);
119    }
120
121    private String[] excludePatterns;
122
123    /**
124     * @parameter expression="${excludes}" alias="excludes"
125     */
126    public void setExcludes( String excludes ) {
127        excludePatterns = splitPatterns(excludes);
128    }
129
130    /**
131     * @parameter expression="${mapper}"
132     */
133    private Mapper mapper;
134
135    /**
136     * @parameter expression="${paths}"
137     */
138    private String[] paths;
139
140    /* For testing only */
141    void setPaths( String[] paths ) {
142        this.paths = paths;
143    }
144
145    public String[] splitPatterns( String patterns ) {
146        String[] result = null;
147        if (patterns != null && patterns.length() > 0) {
148            List<String> tokens = new ArrayList<String>();
149            StringTokenizer tok = new StringTokenizer(patterns, ", ", false);
150            while (tok.hasMoreTokens()) {
151                tokens.add(tok.nextToken());
152            }
153            result = tokens.toArray(new String[tokens.size()]);
154        }
155        return result;
156    }
157
158    public void produce( final DataConsumer pReceiver ) throws IOException {
159        org.vafer.jdeb.mapping.Mapper[] mappers = null;
160        if (mapper != null) {
161            mappers = new org.vafer.jdeb.mapping.Mapper[] { mapper.createMapper() };
162        }
163
164        // link type
165
166        if ("link".equalsIgnoreCase(type)) {
167            if (linkName == null) {
168                throw new RuntimeException("linkName is not set");
169            }
170            if (linkTarget == null) {
171                throw new RuntimeException("linkTarget is not set");
172            }
173
174            new DataProducerLink(linkName, linkTarget, symlink, includePatterns, excludePatterns, mappers).produce(pReceiver);
175            return;
176        }
177
178        // template type
179
180        if ("template".equalsIgnoreCase(type)) {
181            if (paths == null || paths.length == 0) {
182                throw new RuntimeException("paths is not set");
183            }
184
185            new DataProducerPathTemplate(paths, includePatterns, excludePatterns, mappers).produce(pReceiver);
186            return;
187        }
188
189        // Types that require src to exist
190
191        if (src == null || !src.exists()) {
192            if (missingSrc == IGNORE) {
193                return;
194            } else {
195                throw new FileNotFoundException("Data source not found : " + src);
196            }
197        }
198
199        if ("file".equalsIgnoreCase(type)) {
200            new DataProducerFile(src, dst, includePatterns, excludePatterns, mappers).produce(pReceiver);
201            return;
202        }
203
204        if ("archive".equalsIgnoreCase(type)) {
205            new DataProducerArchive(src, includePatterns, excludePatterns, mappers).produce(pReceiver);
206            return;
207        }
208
209        if ("directory".equalsIgnoreCase(type)) {
210            new DataProducerDirectory(src, includePatterns, excludePatterns, mappers).produce(pReceiver);
211            return;
212        }
213
214        throw new IOException("Unknown type '" + type + "' (file|directory|archive|template|link) for " + src);
215    }
216
217}
