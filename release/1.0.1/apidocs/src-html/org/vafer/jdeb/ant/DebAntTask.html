016package org.vafer.jdeb.ant;
017
018import java.io.File;
019import java.io.FileInputStream;
020import java.io.FileOutputStream;
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.Collection;
024
025import org.apache.tools.ant.BuildException;
026import org.apache.tools.ant.taskdefs.MatchingTask;
027import org.apache.tools.ant.taskdefs.Tar;
028import org.apache.tools.ant.types.FileSet;
029import org.vafer.jdeb.Compression;
030import org.vafer.jdeb.DataProducer;
031import org.vafer.jdeb.Processor;
032import org.vafer.jdeb.changes.TextfileChangesProvider;
033import org.vafer.jdeb.descriptors.PackageDescriptor;
034import org.vafer.jdeb.producers.DataProducerFileSet;

036/**
037 * TODO generalize with DebMaker
038 *
039 * AntTask for creating debian archives.
040 * Even supports signed changes files.
041 *
042 * @author Torsten Curdt
043 */
044public class DebAntTask extends MatchingTask {
045
046    /**
047     * The Debian package produced
048     */
049    private File deb;
050
051    /**
052     * The directory containing the control files to build the package
053     */
054    private File control;
055
056    /**
057     * The file containing the PGP keys
058     */
059    private File keyring;
060
061    /**
062     * The key to use in the keyring
063     */
064    private String key;
065
066    /**
067     * The passphrase for the key to sign the changes file
068     */
069    private String passphrase;
070
071    /**
072     * The file to read the changes from
073     */
074    private File changesIn;
075
076    /**
077     * The file where to write the changes to
078     */
079    private File changesOut;
080
081    /**
082     * The file where to write the changes of the changes input to
083     */
084    private File changesSave;
085
086    /**
087     * The compression method used for the data file (none, gzip or bzip2)
088     */
089    private String compression = "gzip";
090
091    /**
092     * Trigger the verbose mode detailing all operations
093     */
094    private boolean verbose;
095
096    private Collection<DataProducer> dataProducers = new ArrayList<DataProducer>();
097
098
099    public void setDestfile( File deb ) {
100        this.deb = deb;
101    }
102
103    public void setControl( File control ) {
104        this.control = control;
105    }
106
107    public void setChangesIn( File changes ) {
108        this.changesIn = changes;
109    }
110
111    public void setChangesOut( File changes ) {
112        this.changesOut = changes;
113    }
114
115    public void setChangesSave( File changes ) {
116        this.changesSave = changes;
117    }
118
119    public void setKeyring( File keyring ) {
120        this.keyring = keyring;
121    }
122
123    public void setKey( String key ) {
124        this.key = key;
125    }
126
127    public void setPassphrase( String passphrase ) {
128        this.passphrase = passphrase;
129    }
130
131    public void setCompression( String compression ) {
132        this.compression = compression;
133    }
134
135    public void setVerbose( boolean verbose ) {
136        this.verbose = verbose;
137    }
138
139    public void addFileSet( FileSet fileset ) {
140        dataProducers.add(new DataProducerFileSet(fileset));
141    }
142
143    public void addTarFileSet( Tar.TarFileSet fileset ) {
144        dataProducers.add(new DataProducerFileSet(fileset));
145    }
146
147    public void addData( Data data ) {
148        dataProducers.add(data);
149    }
150
151    private boolean isPossibleOutput( File file ) {
152
153        if (file == null) {
154            return false;
155        }
156
157        if (file.exists()) {
158            return file.isFile() && file.canWrite();
159        }
160
161        return true;
162    }
163
164    public void execute() {
165
166        if (control == null || !control.isDirectory()) {
167            throw new BuildException("You need to point the 'control' attribute to the control directory.");
168        }
169
170        if (changesIn != null) {
171
172            if (!changesIn.isFile() || !changesIn.canRead()) {
173                throw new BuildException("The 'changesIn' attribute needs to point to a readable file. " + changesIn + " was not found/readable.");
174            }
175
176            if (changesOut == null) {
177                throw new BuildException("A 'changesIn' without a 'changesOut' does not make much sense.");
178            }
179
180            if (!isPossibleOutput(changesOut)) {
181                throw new BuildException("Cannot write the output for 'changesOut' to " + changesOut);
182            }
183
184            if (changesSave != null && !isPossibleOutput(changesSave)) {
185                throw new BuildException("Cannot write the output for 'changesSave' to " + changesSave);
186            }
187
188        } else {
189            if (changesOut != null || changesSave != null) {
190                throw new BuildException("The 'changesOut' or 'changesSave' attributes may only be used when there is a 'changesIn' specified.");
191            }
192        }
193
194        if (Compression.toEnum(compression) == null) {
195            throw new BuildException("The compression method '" + compression + "' is not supported (expected 'none', 'gzip' or 'bzip2')");
196        }
197
198        if (dataProducers.size() == 0) {
199            throw new BuildException("You need to provide at least one reference to a tgz or directory with data.");
200        }
201
202        // validation of the type of the <data> elements
203        for (DataProducer dataProducer : dataProducers) {
204            if (dataProducer instanceof Data) {
205                Data data = (Data) dataProducer;
206                if (data.getType() == null) {
207                    throw new BuildException("The type of the data element wasn't specified (expected 'file', 'directory' or 'archive')");
208                } else if (!Arrays.asList("file", "directory", "archive").contains(data.getType().toLowerCase())) {
209                    throw new BuildException("The type '" + data.getType() + "' of the data element is unknown (expected 'file', 'directory' or 'archive')");
210                }
211            }
212        }
213        
214        if (deb == null) {
215            throw new BuildException("You need to point the 'destfile' attribute to where the deb is supposed to be created.");
216        }
217
218        final File[] controlFiles = control.listFiles();
219
220        final DataProducer[] data = new DataProducer[dataProducers.size()];
221        dataProducers.toArray(data);
222
223        final Processor processor = new Processor(new TaskConsole(this, verbose), null);
224
225        final PackageDescriptor packageDescriptor;
226        try {
227
228            log("Creating debian package: " + deb);
229
230            packageDescriptor = processor.createDeb(controlFiles, data, deb, Compression.toEnum(compression));
231
232        } catch (Exception e) {
233            // what the fuck ant? why are you not printing the exception chain?
234            e.printStackTrace();
235            throw new BuildException("Failed to create debian package " + deb, e);
236        }
237
238        final TextfileChangesProvider changesProvider;
239
240        try {
241            if (changesOut == null) {
242                return;
243            }
244
245            log("Creating changes file: " + changesOut);
246
247            // for now only support reading the changes form a textfile provider
248            changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageDescriptor);
249
250            processor.createChanges(packageDescriptor, changesProvider, (keyring != null) ? new FileInputStream(keyring) : null, key, passphrase, new FileOutputStream(changesOut));
251
252        } catch (Exception e) {
253            throw new BuildException("Failed to create debian changes file " + changesOut, e);
254        }
255
256        try {
257            if (changesSave == null) {
258                return;
259            }
260
261            log("Saving changes to file: " + changesSave);
262
263            changesProvider.save(new FileOutputStream(changesSave));
264
265        } catch (Exception e) {
266            throw new BuildException("Failed to save debian changes file " + changesSave, e);
267        }
268
269    }
270}
