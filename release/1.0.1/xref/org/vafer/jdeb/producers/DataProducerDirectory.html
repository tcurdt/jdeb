16  package org.vafer.jdeb.producers;
17  
18  import java.io.File;
19  import java.io.FileInputStream;
20  import java.io.IOException;
21  import java.io.InputStream;
22  
23  import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
24  import org.apache.tools.ant.DirectoryScanner;
25  import org.vafer.jdeb.DataConsumer;
26  import org.vafer.jdeb.DataProducer;
27  import org.vafer.jdeb.mapping.Mapper;
28  import org.vafer.jdeb.utils.Utils;

30  /**
31   * DataProducer iterating over a directory.
32   * For cross-platform permissions and ownerships you probably want to use a Mapper, too.
33   *
34   * @author Torsten Curdt
35   */
36  public final class DataProducerDirectory extends AbstractDataProducer implements DataProducer {
37  
38      private final DirectoryScanner scanner = new DirectoryScanner();
39  
40      public DataProducerDirectory( final File pDir, final String[] pIncludes, final String[] pExcludes, final Mapper[] pMappers ) {
41          super(pIncludes, pExcludes, pMappers);
42          scanner.setBasedir(pDir);
43          scanner.setIncludes(pIncludes);
44          scanner.setExcludes(pExcludes);
45          scanner.setCaseSensitive(true);
46          scanner.setFollowSymlinks(true);
47      }
48  
49      public void produce( final DataConsumer pReceiver ) throws IOException {
50  
51          scanner.scan();
52  
53          final File baseDir = scanner.getBasedir();
54          
55          for (String dir : scanner.getIncludedDirectories()) {
56              final File file = new File(baseDir, dir);
57              String dirname = getFilename(baseDir, file);
58  
59              if ("".equals(dirname)) {
60                  continue;
61              }
62  
63              if (!isIncluded(dirname)) {
64                  continue;
65              }
66  
67              if ('/' != File.separatorChar) {
68                  dirname = dirname.replace(File.separatorChar, '/');
69              }
70  
71              if (!dirname.endsWith("/")) {
72                  dirname += "/";
73              }
74  
75              TarArchiveEntry entry = new TarArchiveEntry(dirname, true);
76              entry.setUserId(0);
77              entry.setUserName("root");
78              entry.setGroupId(0);
79              entry.setGroupName("root");
80              entry.setMode(TarArchiveEntry.DEFAULT_DIR_MODE);
81  
82              entry = map(entry);
83  
84              entry.setSize(0);
85  
86              pReceiver.onEachDir(entry.getName(), entry.getLinkName(), entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode(), entry.getSize());
87          }
88  
89  
90          for (String f : scanner.getIncludedFiles()) {
91              final File file = new File(baseDir, f);
92              String filename = getFilename(baseDir, file);
93  
94              if (!isIncluded(filename)) {
95                  continue;
96              }
97  
98              if ('/' != File.separatorChar) {
99                  filename = filename.replace(File.separatorChar, '/');
100             }
101 
102             TarArchiveEntry entry = new TarArchiveEntry(filename, true);
103             entry.setUserId(0);
104             entry.setUserName("root");
105             entry.setGroupId(0);
106             entry.setGroupName("root");
107             entry.setMode(TarArchiveEntry.DEFAULT_FILE_MODE);
108 
109             entry = map(entry);
110 
111             entry.setSize(file.length());
112 
113             final InputStream inputStream = new FileInputStream(file);
114             try {
115                 pReceiver.onEachFile(inputStream, entry.getName(), entry.getLinkName(), entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode(), entry.getSize());
116             } finally {
117                 inputStream.close();
118             }
119         }
120     }
121 
122     private String getFilename( File root, File file ) {
123 
124         final String relativeFilename = file.getAbsolutePath().substring(root.getAbsolutePath().length());
125 
126         return Utils.stripLeadingSlash(relativeFilename);
127     }
