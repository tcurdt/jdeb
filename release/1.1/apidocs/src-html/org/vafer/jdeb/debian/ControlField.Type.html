017package org.vafer.jdeb.debian;
018
019import java.io.BufferedReader;
020import java.io.IOException;
021import java.io.StringReader;
022
023/**
024 * A field of a control file. This class is immutable.
025 *
026 * @author Emmanuel Bourg
027 */
028public class ControlField {
029
030    /**
031     * The format of a field.
032     */
033    public enum Type {
034        /** Value on a single line */
035        SIMPLE,
036        /** Value on multiple lines, space characters are ignored */
037        FOLDED,
038        /** Value on multiple lines, space characters are preserved */
039        MULTILINE
040    }
041
042    /** The name of the field */
043    private String name;
044
045    /** Tells if the field is mandatory */
046    private boolean mandatory;
047
048    /** The type of the field */
049    private Type type = Type.SIMPLE;
050
051    /** Tells is the first line of the field must be empty (for MULTILINE values only) */
052    private boolean firstLineEmpty;
053
054
055    public ControlField(String name) {
056        this(name, false);
057    }
058
059    public ControlField(String name, boolean mandatory) {
060        this(name, mandatory, Type.SIMPLE);
061    }
062
063    public ControlField(String name, boolean mandatory, Type type) {
064        this(name, mandatory, type, false);
065    }
066
067    public ControlField(String name, boolean mandatory, Type type, boolean firstLineEmpty) {
068        this.name = name;
069        this.mandatory = mandatory;
070        this.type = type;
071        this.firstLineEmpty = firstLineEmpty;
072    }
073
074    public String getName() {
075        return name;
076    }
077
078    public boolean isMandatory() {
079        return mandatory;
080    }
081
082    public Type getType() {
083        return type;
084    }
085
086    public boolean isFirstLineEmpty() {
087        return firstLineEmpty;
088    }
089
090    /**
091     * Returns the field with the specified value properly formatted. Multiline
092     * values are automatically indented, and dots are added on the empty lines.
093     * 
094     * <pre>
095     * Field-Name: value
096     * </pre>
097     */
098    public String format(String value) {
099        StringBuilder s = new StringBuilder();
100        
101        if (value != null && value.trim().length() > 0) {
102            boolean continuationLine = false;
103            
104            s.append(getName()).append(":");
105            if (isFirstLineEmpty()) {
106                s.append("\n");
107                continuationLine = true;
108            }
109            
110            try {
111                BufferedReader reader = new BufferedReader(new StringReader(value));
112                String line;
113                while ((line = reader.readLine()) != null) {
114                    if (continuationLine && line.trim().length() == 0) {
115                        // put a dot on the empty continuation lines
116                        s.append(" .\n");
117                    } else {
118                        s.append(" ").append(line).append("\n");
119                    }
120
121                    continuationLine = true;
122                }
123            } catch (IOException e) {
124                e.printStackTrace();
125            }
126        }
127        
128        return s.toString();
129    }
130}
