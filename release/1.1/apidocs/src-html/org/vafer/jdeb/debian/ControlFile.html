017package org.vafer.jdeb.debian;
018
019import java.io.BufferedReader;
020import java.io.ByteArrayInputStream;
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.InputStreamReader;
024import java.text.ParseException;
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.HashSet;
028import java.util.LinkedHashMap;
029import java.util.List;
030import java.util.Map;
031import java.util.Set;
032
033/**
034 * A control file as specified by the <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html">Debian policy</a>.
035 *
036 * @author Torsten Curdt
037 */
038public abstract class ControlFile {
039
040    protected final Map<String, String> values = new LinkedHashMap<String, String>();
041    protected final Map<String, String> userDefinedFields = new LinkedHashMap<String, String>();
042    protected final Set<ControlField> userDefinedFieldNames = new HashSet<ControlField>();
043
044    public void parse(String input) throws IOException, ParseException {
045        parse(new ByteArrayInputStream(input.getBytes("UTF-8")));
046    }
047
048    public void parse(InputStream input) throws IOException, ParseException {
049        BufferedReader reader = new BufferedReader(new InputStreamReader(input, "UTF-8"));
050        StringBuilder buffer = new StringBuilder();
051        String field = null;
052        int linenr = 0;
053        while (true) {
054            final String line = reader.readLine();
055
056            if (line == null) {
057                // flush value of the previous field
058                set(field, buffer.toString());
059                break;
060            }
061
062            linenr++;
063
064            if (line.length() == 0) {
065                throw new ParseException("Empty line", linenr);
066            }
067
068            final char first = line.charAt(0);
069            if (Character.isLetter(first)) {
070
071                // new field
072
073                // flush value of the previous field
074                set(field, buffer.toString());
075                buffer = new StringBuilder();
076
077
078                final int i = line.indexOf(':');
079
080                if (i < 0) {
081                    throw new ParseException("Line misses ':' delimiter", linenr);
082                }
083
084                field = line.substring(0, i);
085                buffer.append(line.substring(i + 1).trim());
086
087                continue;
088            }
089
090            // continuing old value, lines with only a dot are ignored
091            buffer.append('\n');
092            if (!".".equals(line.substring(1).trim())) {
093                buffer.append(line.substring(1));
094            }
095        }
096        reader.close();
097
098    }
099
100    public void set(String field, final String value) {
101        if (field != null && isUserDefinedField(field)) {
102            userDefinedFields.put(field, value);
103            String fieldName = getUserDefinedFieldName(field);
104
105            if (fieldName != null) {
106                userDefinedFieldNames.add(new ControlField(fieldName));
107            }
108
109            field = fieldName;
110        }
111
112        if (field != null && !"".equals(field)) {
113            values.put(field, value);
114        }
115    }
116
117    public String get(String field) {
118        return values.get(field);
119    }
120
121    protected abstract ControlField[] getFields();
122
123    protected Map<String, String> getUserDefinedFields() {
124        return userDefinedFields;
125    }
126
127    protected Set<ControlField> getUserDefinedFieldNames() {
128        return userDefinedFieldNames;
129    }
130
131    public List<String> getMandatoryFields() {
132        List<String> fields = new ArrayList<String>();
133
134        for (ControlField field : getFields()) {
135            if (field.isMandatory()) {
136                fields.add(field.getName());
137            }
138        }
139
140        return fields;
141    }
142
143    public boolean isValid() {
144        return invalidFields().size() == 0;
145    }
146
147    public Set<String> invalidFields() {
148        Set<String> invalid = new HashSet<String>();
149
150        for (ControlField field : getFields()) {
151            if (field.isMandatory() && get(field.getName()) == null) {
152                invalid.add(field.getName());
153            }
154        }
155
156        return invalid;
157    }
158
159    public String toString(ControlField... fields) {
160        StringBuilder s = new StringBuilder();
161        for (ControlField field : fields) {
162            String value = values.get(field.getName());
163            s.append(field.format(value));
164        }
165        return s.toString();
166    }
167
168    public String toString() {
169        List<ControlField> fields = new ArrayList<ControlField>();
170        fields.addAll(Arrays.asList(getFields()));
171        fields.addAll(getUserDefinedFieldNames());
172        return toString(fields.toArray(new ControlField[fields.size()]));
173    }
174
175    /**
176     * Returns the letter expected in the prefix of a user defined field
177     * in order to include the field in this control file.
178     *
179     * @return The letter returned is:
180     * <ul>
181     *   <li>B: for a binary package</li>
182     *   <li>S: for a source package</li>
183     *   <li>C: for a changes file</li>
184     * </ul>
185     * 
186     * @since 1.1
187     * @see <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html#s5.7">Debian Policy - User-defined fields</a>
188     */
189    protected abstract char getUserDefinedFieldLetter();
190
191    /**
192     * Tells if the specified field name is a user defined field.
193     * User-defined fields must begin with an 'X', followed by one or more
194     * letters that specify the output file and a hyphen.
195     * 
196     * @param field the name of the field
197     *
198     * @since 1.1
199     * @see <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html#s5.7">Debian Policy - User-defined fields</a>
200     */
201    protected boolean isUserDefinedField(String field) {
202        return field.startsWith("X") && field.indexOf("-") > 0;
203    }
204
205    /**
206     * Returns the user defined field without its prefix.
207     * 
208     * @param field the name of the user defined field
209     * @return the user defined field without the prefix, or null if the fields
210     *         doesn't apply to this control file.
211     * @since 1.1
212     */
213    protected String getUserDefinedFieldName(String field) {
214        int index = field.indexOf('-');
215        char letter = getUserDefinedFieldLetter();
216
217        for (int i = 0; i < index; ++i) {
218            if (field.charAt(i) == letter) {
219                return field.substring(index + 1);
220            }
221        }
222
223        return null;
224    }
225}
