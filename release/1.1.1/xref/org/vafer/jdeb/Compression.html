17  package org.vafer.jdeb;
18  
19  import java.io.OutputStream;
20  
21  import org.apache.commons.compress.compressors.CompressorException;
22  import org.apache.commons.compress.compressors.CompressorStreamFactory;
23  
24  /**
25   * Compression method used for the data file.
26   * 
27   * @author Emmanuel Bourg
28   */
29  public enum Compression {
30  
31      NONE(""),
32      GZIP(".gz"),
33      BZIP2(".bz2"),
34      XZ(".xz");
35  
36      private String extension;
37  
38      private Compression(String extension) {
39          this.extension = extension;
40      }
41  
42      /**
43       * Returns the extension of the compression method
44       */
45      public String getExtension() {
46          return extension;
47      }
48  
49      public OutputStream toCompressedOutputStream(OutputStream out) throws CompressorException {
50          switch (this) {
51              case GZIP:
52                  return new CompressorStreamFactory().createCompressorOutputStream("gz", out);
53              case BZIP2:
54                  return new CompressorStreamFactory().createCompressorOutputStream("bzip2", out);
55              case XZ:
56                  return new CompressorStreamFactory().createCompressorOutputStream("xz", out);
57              default:
58                  return out;
59          }
60      }
61  
62      /**
63       * Returns the compression method corresponding to the specified name.
64       * The matching is case insensitive.
65       * 
66       * @param name the name of the compression method
67       * @return the compression method, or null if not recognized
68       */
69      public static Compression toEnum(String name) {
70          if ("gzip".equalsIgnoreCase(name) || "gz".equalsIgnoreCase(name)) {
71              return GZIP;
72          } else if ("bzip2".equalsIgnoreCase(name) || "bz2".equalsIgnoreCase(name)) {
73              return BZIP2;
74          } else if ("xz".equalsIgnoreCase(name)) {
75              return XZ;
76          } else if ("none".equalsIgnoreCase(name)) {
77              return NONE;
78          } else {
79              return null;
80          }
81      }
82  }
