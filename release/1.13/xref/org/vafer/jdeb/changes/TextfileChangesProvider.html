37  /**
38   * Gets the changes from a changes file. The first entry are the current changes.
39   * The release line will be added. Example:
40   *
41   * release date=22:13 19.08.2007,version=1.5+r90114,urgency=low,by=Torsten Curdt &lt;torsten@vafer.org&gt;
42   *   * debian changes support
43   * release date=20:13 17.08.2007,version=1.4+r89114,urgency=low,by=Torsten Curdt &lt;torsten@vafer.org&gt;
44   *   * debian changes support
45   *
46   */
47  public final class TextfileChangesProvider implements ChangesProvider {
48  
49      private final ChangeSet[] changeSets;
50  
51      private final DateFormat fmt;
52

55          this(pInput, packageControlFile, null, Charset.defaultCharset());

67      public TextfileChangesProvider( final InputStream pInput, final BinaryPackageControlFile packageControlFile, final Long outputTimestampMs, final Charset encoding ) throws IOException, ParseException {
68  
69          fmt = new SimpleDateFormat("HH:mm dd.MM.yyyy");
70          if (outputTimestampMs != null) {
71              fmt.setTimeZone(TimeZone.getTimeZone("UTC"));
72          }
73  
74          final BufferedReader reader = new BufferedReader(new InputStreamReader(pInput, encoding));
75  
76          String packageName = packageControlFile.get("Package");
77          String version = packageControlFile.get("Version");
78          Date date = outputTimestampMs == null ? new Date() : new Date(outputTimestampMs);
79          String distribution = packageControlFile.get("Distribution");
80          String urgency = packageControlFile.get("Urgency");
81          String changedBy = packageControlFile.get("Maintainer");
82          Collection<String> changesColl = new ArrayList<>();
83          Collection<ChangeSet> changeSetColl = new ArrayList<>();
84  
85  
86          while (true) {
87              final String line = reader.readLine();
88              if (line == null) {
89                  final String[] changes = changesColl.toArray(new String[changesColl.size()]);
90                  final ChangeSet changeSet = new ChangeSet(packageName, version, date, distribution, urgency, changedBy, changes);
91                  changeSetColl.add(changeSet);
92                  break;
93              }
94  
95              if (line.startsWith("release ")) {
96  
97                  if (changesColl.size() > 0) {
98                      final String[] changes = changesColl.toArray(new String[changesColl.size()]);
99                      final ChangeSet changeSet = new ChangeSet(packageName, version, date, distribution, urgency, changedBy, changes);
100                     changeSetColl.add(changeSet);
101                     changesColl.clear();
102                 }
103 
104                 final String[] tokens = line.substring("release ".length()).split(",");
105                 for (String token : tokens) {
106                     final String[] lr = token.trim().split("=");
107                     final String key = lr[0];
108                     final String value = lr[1];
109 
110                     if ("urgency".equals(key)) {
111                         urgency = value;
112                     } else if ("by".equals(key)) {
113                         changedBy = value;
114                     } else if ("date".equals(key)) {
115                         date = fmt.parse(value);
116                     } else if ("version".equals(key)) {
117                         version = value;
118                     } else if ("distribution".equals(key)) {
119                         distribution = value;
120                     }
121                 }
122                 continue;
123             }
124 
125             if (line.startsWith(" * ")) {
126                 changesColl.add(line.substring(" * ".length()));
127                 continue;
128             }
129 
130             throw new ParseException("Unknown line syntax [" + line + "]", 0);
131         }
132 
133         reader.close();
134 
135         changeSets = changeSetColl.toArray(new ChangeSet[changeSetColl.size()]);
136     }
137 
138     public void save(OutputStream pOutput) throws IOException {
139         BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(pOutput));
140 
141         for (ChangeSet changeSet : changeSets) {
142             writer.write("release ");
143             writer.write("date=" + fmt.format(changeSet.getDate()) + ",");
144             writer.write("version=" + changeSet.getVersion() + ",");
145             writer.write("urgency=" + changeSet.getUrgency() + ",");
146             writer.write("by=" + changeSet.getChangedBy() + ",");
147             writer.write("distribution=" + changeSet.getDistribution());
148             writer.write("\n");
149 
150             for (String change : changeSet.getChanges()) {
151                 writer.write(" * ");
152                 writer.write(change);
153                 writer.write("\n");
154             }
155         }
156 
157         writer.close();
158     }
159 
160     public ChangeSet[] getChangesSets() {
161         return changeSets;
162     }
163 }
