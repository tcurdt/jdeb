10  public class OutputTimestampResolver {
11      private final Console console;
12      private final EnvironmentVariablesReader envReader;
13  
14      public OutputTimestampResolver(Console console) {
15          this(console, new EnvironmentVariablesReader());
16      }
17  
18      OutputTimestampResolver(Console console, EnvironmentVariablesReader envReader) {
19          this.console = console;
20          this.envReader = envReader;
21      }
22  
23      public Long resolveOutputTimestamp(String paramValue) {
24          if (paramValue != null) {
25              Optional<Instant> outputDate = MavenArchiver.parseBuildOutputTimestamp(paramValue);
26              if (outputDate.isPresent()) {
27                  console.info("Accepted outputTimestamp parameter: " + paramValue);
28                  return outputDate.get().toEpochMilli();
29              }
30          }
31  
32          String sourceDate = envReader.getSourceDateEpoch();
33          if (sourceDate != null && !sourceDate.isEmpty()) {
34              try {
35                  long sourceDateVal = Long.parseLong(sourceDate);
36                  console.info("Accepted SOURCE_DATE_EPOCH environment variable: " + sourceDate);
37                  return sourceDateVal * TimeUnit.SECONDS.toMillis(1);
38              } catch (NumberFormatException e) {
39                  throw new IllegalArgumentException("Invalid SOURCE_DATE_EPOCH environment variable value: " + sourceDate, e);
40              }
41          }
42  
43          return null;
44      }
45  
46      static class EnvironmentVariablesReader {
47          String getSourceDateEpoch() {
48              return System.getenv("SOURCE_DATE_EPOCH");
49          }
50      }
51  }
