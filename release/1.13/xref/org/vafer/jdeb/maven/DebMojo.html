403     protected void setData( Data[] dataSet ) {
404         this.dataSet = dataSet;
405         dataProducers.clear();
406         conffileProducers.clear();
407         if (dataSet != null) {
408             Collections.addAll(dataProducers, dataSet);
409

411                 if (item.getConffile()) {
412                     conffileProducers.add(item);

414             }

418     @SuppressWarnings("unchecked,rawtypes")
419     protected VariableResolver initializeVariableResolver( Map<String, String> variables, Long outputTimestampMs ) {
420         variables.putAll((Map) getProject().getProperties());
421         variables.putAll((Map) System.getProperties());
422         variables.put("name", name != null ? name : getProject().getName());
423         variables.put("artifactId", getProject().getArtifactId());
424         variables.put("groupId", getProject().getGroupId());
425         variables.put("version", getProjectVersion(outputTimestampMs));
426         variables.put("description", getProject().getDescription());
427         variables.put("extension", "deb");
428         variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
429         variables.put("buildDir", buildDirectory.getAbsolutePath());
430         variables.put("project.version", getProject().getVersion());
431 
432         if (getProject().getInceptionYear() != null) {
433             variables.put("project.inceptionYear", getProject().getInceptionYear());
434         }
435         if (getProject().getOrganization() != null) {
436             if (getProject().getOrganization().getName() != null) {
437                 variables.put("project.organization.name", getProject().getOrganization().getName());
438             }
439             if (getProject().getOrganization().getUrl() != null) {
440                 variables.put("project.organization.url", getProject().getOrganization().getUrl());
441             }
442         }
443 
444         variables.put("url", getProject().getUrl());
445 
446         return new MapVariableResolver(variables);
447     }

495     /**
496      * Main entry point
497      *
498      * @throws MojoExecutionException on error
499      */
500     public void execute() throws MojoExecutionException {
501 
502         final MavenProject project = getProject();
503 
504         if (skip) {
505             getLog().info("skipping as configured (skip)");
506             return;
507         }
508 
509         if (skipPOMs && isPOM()) {
510             getLog().info("skipping because artifact is a pom (skipPOMs)");
511             return;
512         }
513 
514         if (skipSubmodules && isSubmodule()) {
515             getLog().info("skipping submodule (skipSubmodules)");
516             return;
517         }
518 
519 
520         setData(dataSet);
521 
522         console = new MojoConsole(getLog(), verbose);
523 
524         initializeSignProperties();
525         
526         Long outputTimestampMs = new OutputTimestampResolver(console).resolveOutputTimestamp(outputTimestamp);
527 
528         final VariableResolver resolver = initializeVariableResolver(new HashMap<String, String>(), outputTimestampMs);
529 
530         final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
531         final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
532         final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
533         final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
534         final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
535         final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
536         final File keyringFile = keyring == null ? null : new File(Utils.replaceVariables(resolver, keyring, openReplaceToken, closeReplaceToken));
537 
538         // if there are no producers defined we try to use the artifacts
539         if (dataProducers.isEmpty()) {
540 
541             if (hasMainArtifact()) {
542                 Set<Artifact> artifacts = new HashSet<>();
543 
544                 artifacts.add(project.getArtifact());
545 
546                 @SuppressWarnings("unchecked")
547                 final Set<Artifact> projectArtifacts = project.getArtifacts();
548 
549                 artifacts.addAll(projectArtifacts);
550 
551                 @SuppressWarnings("unchecked")
552                 final List<Artifact> attachedArtifacts = project.getAttachedArtifacts();
553 
554                 artifacts.addAll(attachedArtifacts);
555 
556                 for (Artifact artifact : artifacts) {
557                     final File file = artifact.getFile();
558                     if (file != null) {
559                         dataProducers.add(new DataProducer() {
560                             public void produce( final DataConsumer receiver ) {
561                                 try {
562                                     final File path = new File(installDirFile.getPath(), file.getName());
563                                     final String entryName = path.getPath();
564 
565                                     final boolean symbolicLink = SymlinkUtils.isSymbolicLink(path);
566                                     final TarArchiveEntry e;
567                                     if (symbolicLink) {
568                                         e = new TarArchiveEntry(entryName, TarConstants.LF_SYMLINK);
569                                         e.setLinkName(SymlinkUtils.readSymbolicLink(path));
570                                     } else {
571                                         e = new TarArchiveEntry(entryName, true);
572                                     }
573 
574                                     e.setUserId(0);
575                                     e.setGroupId(0);
576                                     e.setUserName("root");
577                                     e.setGroupName("root");
578                                     e.setMode(TarEntry.DEFAULT_FILE_MODE);
579                                     e.setSize(file.length());
580 
581                                     receiver.onEachFile(new FileInputStream(file), e);
582                                 } catch (Exception e) {
583                                     getLog().error(e);
584                                 }
585                             }
586                         });
587                     } else {
588                         getLog().error("No file for artifact " + artifact);
589                     }
590                 }
591             }
592         }
593 
594         try {
595             DebMaker debMaker = new DebMaker(console, dataProducers, conffileProducers);
596             debMaker.setDeb(debFile);
597             debMaker.setControl(controlDirFile);
598             debMaker.setEncoding(encoding);
599             debMaker.setPackage(getProject().getArtifactId());
600             debMaker.setDescription(getProject().getDescription());
601             debMaker.setHomepage(getProject().getUrl());
602             debMaker.setChangesIn(changesInFile);
603             debMaker.setChangesOut(changesOutFile);
604             debMaker.setChangesSave(changesSaveFile);
605             debMaker.setChangesEnabled(changesEnabled);
606             debMaker.setCompression(compression);
607             debMaker.setKeyring(keyringFile);
608             debMaker.setKey(key);
609             debMaker.setPassphrase(passphrase);
610             debMaker.setSignPackage(signPackage);
611             debMaker.setSignChanges(signChanges);
612             debMaker.setSignMethod(signMethod);
613             debMaker.setSignRole(signRole);
614             debMaker.setResolver(resolver);
615             debMaker.setOpenReplaceToken(openReplaceToken);
616             debMaker.setCloseReplaceToken(closeReplaceToken);
617             debMaker.setDigest(digest);
618             debMaker.setTarBigNumberMode(tarBigNumberMode);
619             debMaker.setTarLongFileMode(tarLongFileMode);
620             debMaker.setIgnoreBrokenLinks(ignoreBrokenLinks);
621             debMaker.setOutputTimestampMs(outputTimestampMs);
622             debMaker.validate();
623             debMaker.makeDeb();
624 
625             // Always attach unless explicitly set to false
626             if ("true".equalsIgnoreCase(attach)) {
627                 console.info("Attaching created debian package " + debFile);
628                 if (!isType()) {
629                     projectHelper.attachArtifact(project, type, classifier, debFile);
630                 } else {
631                     project.getArtifact().setFile(debFile);
632                 }
633             }
634 
635         } catch (PackagingException e) {
636             getLog().error("Failed to create debian package " + debFile, e);
637             throw new MojoExecutionException("Failed to create debian package " + debFile, e);
638         }
639 
640         if (!isBlank(propertyPrefix)) {
641           project.getProperties().put(propertyPrefix+"version", getProjectVersion(outputTimestampMs) );
642           project.getProperties().put(propertyPrefix+"deb", debFile.getAbsolutePath());
643           project.getProperties().put(propertyPrefix+"deb.name", debFile.getName());
644           project.getProperties().put(propertyPrefix+"changes", changesOutFile.getAbsolutePath());
645           project.getProperties().put(propertyPrefix+"changes.name", changesOutFile.getName());
646           project.getProperties().put(propertyPrefix+"changes.txt", changesSaveFile.getAbsolutePath());
647           project.getProperties().put(propertyPrefix+"changes.txt.name", changesSaveFile.getName());
648         }
649 
650     }

652     /**
653      * Initializes unspecified sign properties using available defaults
654      * and global settings.
655      */
656     private void initializeSignProperties() {
657         if (!signPackage && !(signChanges && changesEnabled)) {
658             return;
659         }
660 
661         if (key != null && keyring != null && passphrase != null) {
662             return;
663         }
664 
665         Map<String, String> properties =
666                 readPropertiesFromActiveProfiles(signCfgPrefix, KEY, KEYRING, PASSPHRASE);
667 
668         key = lookupIfEmpty(key, properties, KEY);
669         keyring = lookupIfEmpty(keyring, properties, KEYRING);
670         passphrase = decrypt(lookupIfEmpty(passphrase, properties, PASSPHRASE));
671 
672         if (keyring == null) {
673             try {
674                 keyring = Utils.guessKeyRingFile().getAbsolutePath();
675                 console.info("Located keyring at " + keyring);
676             } catch (FileNotFoundException e) {
677                 console.warn(e.getMessage());
678             }
679         }
680     }

682     /**
683      * Decrypts given passphrase if needed using maven security dispatcher.
684      * See http://maven.apache.org/guides/mini/guide-encryption.html for details.
685      *
686      * @param maybeEncryptedPassphrase possibly encrypted passphrase
687      * @return decrypted passphrase
688      */
689     private String decrypt( final String maybeEncryptedPassphrase ) {
690         if (maybeEncryptedPassphrase == null) {
691             return null;
692         }
693 
694         try {
695             final String decrypted = secDispatcher.decrypt(maybeEncryptedPassphrase);
696             if (maybeEncryptedPassphrase.equals(decrypted)) {
697                 console.info("Passphrase was not encrypted");
698             } else {
699                 console.info("Passphrase was successfully decrypted");
700             }
701             return decrypted;
702         } catch (SecDispatcherException e) {
703             console.warn("Unable to decrypt passphrase: " + e.getMessage());
704         }
705 
706         return maybeEncryptedPassphrase;
707     }

723     /**
724      * Read properties from the active profiles.
725      *
726      * Goes through all active profiles (in the order the
727      * profiles are defined in settings.xml) and extracts
728      * the desired properties (if present). The prefix is
729      * used when looking up properties in the profile but
730      * not in the returned map.
731      *
732      * @param prefix The prefix to use or null if no prefix should be used
733      * @param properties The properties to read
734      *
735      * @return A map containing the values for the properties that were found
736      */
737     public Map<String, String> readPropertiesFromActiveProfiles( final String prefix,
738                                                                  final String... properties ) {
739         if (settings == null) {
740             console.debug("No maven setting injected");
741             return Collections.emptyMap();
742         }
743 
744         final List<String> activeProfilesList = settings.getActiveProfiles();
745         if (activeProfilesList.isEmpty()) {
746             console.debug("No active profiles found");
747             return Collections.emptyMap();
748         }
749 
750         final Map<String, String> map = new HashMap<>();
751         final Set<String> activeProfiles = new HashSet<>(activeProfilesList);
752 
753         // Iterate over all active profiles in order
754         for (final Profile profile : settings.getProfiles()) {
755             // Check if the profile is active
756             final String profileId = profile.getId();
757             if (activeProfiles.contains(profileId)) {
758                 console.debug("Trying active profile " + profileId);
759                 for (final String property : properties) {
760                     final String propKey = prefix != null ? prefix + property : property;
761                     final String value = profile.getProperties().getProperty(propKey);
762                     if (value != null) {
763                         console.debug("Found property " + property + " in profile " + profileId);
764                         map.put(property, value);
765                     }
766                 }
767             }
768         }
769 
770         return map;
771     }
