16  package org.vafer.jdeb.producers;
17  
18  import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
19  
20  import org.apache.commons.io.IOUtils;
21  import org.vafer.jdeb.DataConsumer;
22  
23  import java.io.IOException;
24  import java.io.InputStream;
25  
26  /**
27   * Package-private utility class with common producers functionality.
28   *
29   * @author Roman Kashitsyn <roman.kashitsyn@gmail.com>
30   */
31  class Producers {
32  
33      final static int ROOT_UID = 0;
34      final static String ROOT_NAME = "root";
35  
36      private Producers() {}
37  
38  
39      /**
40       * Creates a tar file entry with defaults parameters.
41       * @param entryName the entry name
42       * @return file entry with reasonable defaults
43       */
44      static TarArchiveEntry defaultFileEntryWithName( final String entryName ) {
45          TarArchiveEntry entry = new TarArchiveEntry(entryName, true);
46          entry.setUserId(ROOT_UID);
47          entry.setUserName(ROOT_NAME);
48          entry.setGroupId(ROOT_UID);
49          entry.setGroupName(ROOT_NAME);
50          entry.setMode(TarArchiveEntry.DEFAULT_FILE_MODE);
51          return entry;
52      }
53  
54      /**
55       * Creates a tar directory entry with defaults parameters.
56       * @param dirName the directory name
57       * @return dir entry with reasonable defaults
58       */
59      static TarArchiveEntry defaultDirEntryWithName( final String dirName ) {
60          TarArchiveEntry entry = new TarArchiveEntry(dirName, true);
61          entry.setUserId(ROOT_UID);
62          entry.setUserName(ROOT_NAME);
63          entry.setGroupId(ROOT_UID);
64          entry.setGroupName(ROOT_NAME);
65          entry.setMode(TarArchiveEntry.DEFAULT_DIR_MODE);
66          return entry;
67      }
68  
69      /**
70       * Forwards tar archive entry entry to a consumer.
71       * @param consumer the consumer
72       * @param entry the entry to pass
73       * @throws IOException
74       */
75      static void produceDirEntry( final DataConsumer consumer,
76                                   final TarArchiveEntry entry ) throws IOException {
77          consumer.onEachDir(
78                  entry.getName(),
79                  entry.getLinkName(),
80                  entry.getUserName(),
81                  entry.getUserId(),
82                  entry.getGroupName(),
83                  entry.getGroupId(),
84                  entry.getMode(),
85                  entry.getSize()
86          );
87      }
88  
89  
90      /**
91       * Feeds input stream to data consumer using metadata from tar entry.
92       * @param consumer the consumer
93       * @param inputStream the stream to feed
94       * @param entry the entry to use for metadata
95       * @throws IOException on consume error
96       */
97      static void produceInputStreamWithEntry( final DataConsumer consumer,
98                                               final InputStream inputStream,
99                                               final TarArchiveEntry entry ) throws IOException {
100         try {
101             consumer.onEachFile(inputStream,
102                     entry.getName(),
103                     entry.getLinkName(),
104                     entry.getUserName(),
105                     entry.getUserId(),
106                     entry.getGroupName(),
107                     entry.getGroupId(),
108                     entry.getMode(),
109                     entry.getSize()
110             );
111         } finally {
112             IOUtils.closeQuietly(inputStream);
113         }
114     }
115 
116 }
