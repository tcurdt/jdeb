017package org.vafer.jdeb.signing;
018
019import java.io.ByteArrayInputStream;
020import java.io.IOException;
021import java.io.InputStream;
022import java.io.InputStreamReader;
023import java.io.OutputStream;
024import java.nio.charset.Charset;
025import java.security.GeneralSecurityException;
026import java.util.Iterator;
027
028import org.apache.commons.io.LineIterator;
029import org.bouncycastle.bcpg.ArmoredOutputStream;
030import org.bouncycastle.bcpg.BCPGOutputStream;
031import org.bouncycastle.openpgp.PGPException;
032import org.bouncycastle.openpgp.PGPPrivateKey;
033import org.bouncycastle.openpgp.PGPSecretKey;
034import org.bouncycastle.openpgp.PGPSecretKeyRing;
035import org.bouncycastle.openpgp.PGPSecretKeyRingCollection;
036import org.bouncycastle.openpgp.PGPSignature;
037import org.bouncycastle.openpgp.PGPSignatureGenerator;
038import org.bouncycastle.openpgp.PGPUtil;
039import org.bouncycastle.openpgp.operator.bc.BcPBESecretKeyDecryptorBuilder;
040import org.bouncycastle.openpgp.operator.bc.BcPGPContentSignerBuilder;
041import org.bouncycastle.openpgp.operator.bc.BcPGPDigestCalculatorProvider;
042
043/**
044 * Signing with OpenPGP.
045 * 
046 * @author Torsten Curdt
047 * @author Emmanuel Bourg
048 */
049public class PGPSigner {
050
051    private static final byte[] EOL = "\n".getBytes(Charset.forName("UTF-8"));
052
053    private PGPSecretKey secretKey;
054    private PGPPrivateKey privateKey;
055
056    public PGPSigner(InputStream keyring, String keyId, String passphrase) throws IOException, PGPException {
057        secretKey = getSecretKey(keyring, keyId);
058        if(secretKey == null)
059        {
060            throw new PGPException(String.format("Specified key %s does not exist in key ring %s", keyId, keyring));
061        }
062        privateKey = secretKey.extractPrivateKey(new BcPBESecretKeyDecryptorBuilder(new BcPGPDigestCalculatorProvider()).build(passphrase.toCharArray()));
063    }
064
065    /**
066     * Creates a clear sign signature over the input data. (Not detached)
067     *
068     * @param input      the content to be signed
069     * @param output     the output destination of the signature
070     */
071    public void clearSign(String input, OutputStream output) throws IOException, PGPException, GeneralSecurityException {
072        clearSign(new ByteArrayInputStream(input.getBytes("UTF-8")), output);
073    }
074
075    /**
076     * Creates a clear sign signature over the input data. (Not detached)
077     *
078     * @param input      the content to be signed
079     * @param output     the output destination of the signature
080     */
081    public void clearSign(InputStream input, OutputStream output) throws IOException, PGPException, GeneralSecurityException {
082        int digest = PGPUtil.SHA1;
083        
084        PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(privateKey.getPublicKeyPacket().getAlgorithm(), digest));
085        signatureGenerator.init(PGPSignature.CANONICAL_TEXT_DOCUMENT, privateKey);
086        
087        ArmoredOutputStream armoredOutput = new ArmoredOutputStream(output);
088        armoredOutput.beginClearText(digest);
089        
090        LineIterator iterator = new LineIterator(new InputStreamReader(input));
091        
092        while (iterator.hasNext()) {
093            String line = iterator.nextLine();
094            
095            // trailing spaces must be removed for signature calculation (see http://tools.ietf.org/html/rfc4880#section-7.1)
096            byte[] data = trim(line).getBytes("UTF-8");
097            
098            armoredOutput.write(data);
099            armoredOutput.write(EOL);
100            
101            signatureGenerator.update(data);
102            if (iterator.hasNext()) {
103                signatureGenerator.update(EOL);
104            }
105        }
106
107        armoredOutput.endClearText();
108        
109        PGPSignature signature = signatureGenerator.generate();
110        signature.encode(new BCPGOutputStream(armoredOutput));
111        
112        armoredOutput.close();
113    }
114    
115    /**
116     * Returns the secret key.
117     */
118    public PGPSecretKey getSecretKey()
119    {
120        return secretKey;
121    }
122
123    /**
124     * Returns the private key.
125     */
126    public PGPPrivateKey getPrivateKey()
127    {
128        return privateKey;
129    }
130
131    /**
132     * Returns the secret key matching the specified identifier.
133     * 
134     * @param input the input stream containing the keyring collection
135     * @param keyId the 4 bytes identifier of the key
136     */
137    private PGPSecretKey getSecretKey(InputStream input, String keyId) throws IOException, PGPException {
138        PGPSecretKeyRingCollection keyrings = new PGPSecretKeyRingCollection(PGPUtil.getDecoderStream(input));
139        
140        Iterator rIt = keyrings.getKeyRings();
141
142        while (rIt.hasNext()) {
143            PGPSecretKeyRing kRing = (PGPSecretKeyRing) rIt.next();
144            Iterator kIt = kRing.getSecretKeys();
145
146            while (kIt.hasNext()) {
147                PGPSecretKey key = (PGPSecretKey) kIt.next();
148                
149                if (key.isSigningKey() && Long.toHexString(key.getKeyID() & 0xFFFFFFFFL).equals(keyId.toLowerCase())) {
150                    return key;
151                }
152            }
153        }
154
155        return null;
156    }
157
158    /**
159     * Trim the trailing spaces.
160     * 
161     * @param line
162     */
163    private String trim(String line) {
164        char[] chars = line.toCharArray();
165        int len = chars.length;
166
167        while (len > 0) {
168            if (!Character.isWhitespace(chars[len - 1])) {
169                break;
170            }
171            len--;
172        }
173        
174        return line.substring(0, len);
175    }
176}
