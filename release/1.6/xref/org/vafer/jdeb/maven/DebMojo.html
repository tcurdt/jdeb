375     protected void setData( Data[] dataSet ) {
376         this.dataSet = dataSet;
377         dataProducers.clear();
378         conffileProducers.clear();
379         if (dataSet != null) {
380             Collections.addAll(dataProducers, dataSet);
381 
382             for (Data item : dataSet) {
383                 if (item.getConffile()) {
384                     conffileProducers.add(item);
385                 }
386             }
387         }

390     protected VariableResolver initializeVariableResolver( Map<String, String> variables ) {
391         @SuppressWarnings("unchecked")
392         final Map<String, String> projectProperties = Map.class.cast(getProject().getProperties());
393         @SuppressWarnings("unchecked")
394         final Map<String, String> systemProperties = Map.class.cast(System.getProperties());
395 
396         variables.putAll(projectProperties);
397         variables.putAll(systemProperties);
398         variables.put("name", name != null ? name : getProject().getName());
399         variables.put("artifactId", getProject().getArtifactId());
400         variables.put("groupId", getProject().getGroupId());
401         variables.put("version", getProjectVersion());
402         variables.put("description", getProject().getDescription());
403         variables.put("extension", "deb");
404         variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
405         variables.put("buildDir", buildDirectory.getAbsolutePath());
406         variables.put("project.version", getProject().getVersion());
407         variables.put("url", getProject().getUrl());
408 
409         return new MapVariableResolver(variables);
410     }

457     /**
458      * Main entry point
459      *
460      * @throws MojoExecutionException on error
461      */
462     public void execute() throws MojoExecutionException {
463 
464         final MavenProject project = getProject();
465 
466         if (skip) {
467             getLog().info("skipping as configured (skip)");
468             return;
469         }
470 
471         if (skipPOMs && isPOM()) {
472             getLog().info("skipping because artifact is a pom (skipPOMs)");
473             return;
474         }
475 
476         if (skipSubmodules && isSubmodule()) {
477             getLog().info("skipping submodule (skipSubmodules)");
478             return;
479         }
480 
481 
482         setData(dataSet);
483 
484         console = new MojoConsole(getLog(), verbose);
485 
486         initializeSignProperties();
487 
488         final VariableResolver resolver = initializeVariableResolver(new HashMap<String, String>());
489 
490         final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
491         final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
492         final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
493         final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
494         final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
495         final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
496         final File keyringFile = keyring == null ? null : new File(Utils.replaceVariables(resolver, keyring, openReplaceToken, closeReplaceToken));
497 
498         // if there are no producers defined we try to use the artifacts
499         if (dataProducers.isEmpty()) {
500 
501             if (hasMainArtifact()) {
502                 Set<Artifact> artifacts = new HashSet<Artifact>();
503 
504                 artifacts.add(project.getArtifact());
505 
506                 @SuppressWarnings("unchecked")
507                 final Set<Artifact> projectArtifacts = project.getArtifacts();
508 
509                 for (Artifact artifact : projectArtifacts) {
510                     artifacts.add(artifact);
511                 }
512 
513                 @SuppressWarnings("unchecked")
514                 final List<Artifact> attachedArtifacts = project.getAttachedArtifacts();
515 
516                 for (Artifact artifact : attachedArtifacts) {
517                     artifacts.add(artifact);
518                 }
519 
520                 for (Artifact artifact : artifacts) {
521                     final File file = artifact.getFile();
522                     if (file != null) {
523                         dataProducers.add(new DataProducer() {
524                             public void produce( final DataConsumer receiver ) {
525                                 try {
526                                     final File path = new File(installDirFile.getPath(), file.getName());
527                                     final String entryName = path.getPath();
528 
529                                     final boolean symbolicLink = SymlinkUtils.isSymbolicLink(path);
530                                     final TarArchiveEntry e;
531                                     if (symbolicLink) {
532                                         e = new TarArchiveEntry(entryName, TarConstants.LF_SYMLINK);
533                                         e.setLinkName(SymlinkUtils.readSymbolicLink(path));
534                                     } else {
535                                         e = new TarArchiveEntry(entryName, true);
536                                     }
537 
538                                     e.setUserId(0);
539                                     e.setGroupId(0);
540                                     e.setUserName("root");
541                                     e.setGroupName("root");
542                                     e.setMode(TarEntry.DEFAULT_FILE_MODE);
543                                     e.setSize(file.length());
544 
545                                     receiver.onEachFile(new FileInputStream(file), e);
546                                 } catch (Exception e) {
547                                     getLog().error(e);
548                                 }
549                             }
550                         });
551                     } else {
552                         getLog().error("No file for artifact " + artifact);
553                     }
554                 }
555             }
556         }
557 
558         try {
559             DebMaker debMaker = new DebMaker(console, dataProducers, conffileProducers);
560             debMaker.setDeb(debFile);
561             debMaker.setControl(controlDirFile);
562             debMaker.setPackage(getProject().getArtifactId());
563             debMaker.setDescription(getProject().getDescription());
564             debMaker.setHomepage(getProject().getUrl());
565             debMaker.setChangesIn(changesInFile);
566             debMaker.setChangesOut(changesOutFile);
567             debMaker.setChangesSave(changesSaveFile);
568             debMaker.setCompression(compression);
569             debMaker.setKeyring(keyringFile);
570             debMaker.setKey(key);
571             debMaker.setPassphrase(passphrase);
572             debMaker.setSignPackage(signPackage);
573             debMaker.setSignChanges(signChanges);
574             debMaker.setSignMethod(signMethod);
575             debMaker.setSignRole(signRole);
576             debMaker.setResolver(resolver);
577             debMaker.setOpenReplaceToken(openReplaceToken);
578             debMaker.setCloseReplaceToken(closeReplaceToken);
579             debMaker.setDigest(digest);
580             debMaker.setTarBigNumberMode(tarBigNumberMode);
581             debMaker.setTarLongFileMode(tarLongFileMode);
582             debMaker.validate();
583             debMaker.makeDeb();
584 
585             // Always attach unless explicitly set to false
586             if ("true".equalsIgnoreCase(attach)) {
587                 console.info("Attaching created debian package " + debFile);
588                 if (!isType()) {
589                     projectHelper.attachArtifact(project, type, classifier, debFile);
590                 } else {
591                     project.getArtifact().setFile(debFile);
592                 }
593             }
594 
595         } catch (PackagingException e) {
596             getLog().error("Failed to create debian package " + debFile, e);
597             throw new MojoExecutionException("Failed to create debian package " + debFile, e);
598         }
599 
600         if (!isBlank(propertyPrefix)) {
601           project.getProperties().put(propertyPrefix+"version", getProjectVersion() );
602           project.getProperties().put(propertyPrefix+"deb", debFile.getAbsolutePath());
603           project.getProperties().put(propertyPrefix+"deb.name", debFile.getName());
604           project.getProperties().put(propertyPrefix+"changes", changesOutFile.getAbsolutePath());
605           project.getProperties().put(propertyPrefix+"changes.name", changesOutFile.getName());
606           project.getProperties().put(propertyPrefix+"changes.txt", changesSaveFile.getAbsolutePath());
607           project.getProperties().put(propertyPrefix+"changes.txt.name", changesSaveFile.getName());
608         }
609 
610     }

612     /**
613      * Initializes unspecified sign properties using available defaults
614      * and global settings.
615      */
616     private void initializeSignProperties() {
617         if (!signPackage && !signChanges) {
618             return;
619         }
620 
621         if (key != null && keyring != null && passphrase != null) {
622             return;
623         }
624 
625         Map<String, String> properties =
626                 readPropertiesFromActiveProfiles(signCfgPrefix, KEY, KEYRING, PASSPHRASE);
627 
628         key = lookupIfEmpty(key, properties, KEY);
629         keyring = lookupIfEmpty(keyring, properties, KEYRING);
630         passphrase = decrypt(lookupIfEmpty(passphrase, properties, PASSPHRASE));
631 
632         if (keyring == null) {
633             try {
634                 keyring = Utils.guessKeyRingFile().getAbsolutePath();
635                 console.info("Located keyring at " + keyring);
636             } catch (FileNotFoundException e) {
637                 console.warn(e.getMessage());
638             }
639         }
640     }

642     /**
643      * Decrypts given passphrase if needed using maven security dispatcher.
644      * See http://maven.apache.org/guides/mini/guide-encryption.html for details.
645      *
646      * @param maybeEncryptedPassphrase possibly encrypted passphrase
647      * @return decrypted passphrase
648      */
649     private String decrypt( final String maybeEncryptedPassphrase ) {
650         if (maybeEncryptedPassphrase == null) {
651             return null;
652         }
653 
654         try {
655             final String decrypted = secDispatcher.decrypt(maybeEncryptedPassphrase);
656             if (maybeEncryptedPassphrase.equals(decrypted)) {
657                 console.info("Passphrase was not encrypted");
658             } else {
659                 console.info("Passphrase was successfully decrypted");
660             }
661             return decrypted;
662         } catch (SecDispatcherException e) {
663             console.warn("Unable to decrypt passphrase: " + e.getMessage());
664         }
665 
666         return maybeEncryptedPassphrase;
667     }

683     /**
684      * Read properties from the active profiles.
685      *
686      * Goes through all active profiles (in the order the
687      * profiles are defined in settings.xml) and extracts
688      * the desired properties (if present). The prefix is
689      * used when looking up properties in the profile but
690      * not in the returned map.
691      *
692      * @param prefix The prefix to use or null if no prefix should be used
693      * @param properties The properties to read
694      *
695      * @return A map containing the values for the properties that were found
696      */
697     public Map<String, String> readPropertiesFromActiveProfiles( final String prefix,
698                                                                  final String... properties ) {
699         if (settings == null) {
700             console.debug("No maven setting injected");
701             return Collections.emptyMap();
702         }
703 
704         final List<String> activeProfilesList = settings.getActiveProfiles();
705         if (activeProfilesList.isEmpty()) {
706             console.debug("No active profiles found");
707             return Collections.emptyMap();
708         }
709 
710         final Map<String, String> map = new HashMap<String, String>();
711         final Set<String> activeProfiles = new HashSet<String>(activeProfilesList);
712 
713         // Iterate over all active profiles in order
714         for (final Profile profile : settings.getProfiles()) {
715             // Check if the profile is active
716             final String profileId = profile.getId();
717             if (activeProfiles.contains(profileId)) {
718                 console.debug("Trying active profile " + profileId);
719                 for (final String property : properties) {
720                     final String propKey = prefix != null ? prefix + property : property;
721                     final String value = profile.getProperties().getProperty(propKey);
722                     if (value != null) {
723                         console.debug("Found property " + property + " in profile " + profileId);
724                         map.put(property, value);
725                     }
726                 }
727             }
