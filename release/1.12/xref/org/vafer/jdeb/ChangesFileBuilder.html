17  package org.vafer.jdeb;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.security.MessageDigest;
22  import java.security.NoSuchAlgorithmException;
23  import java.util.Date;
24  
25  import org.apache.commons.io.FileUtils;
26  import org.apache.commons.io.output.NullOutputStream;
27  import org.vafer.jdeb.changes.ChangesProvider;
28  import org.vafer.jdeb.debian.BinaryPackageControlFile;
29  import org.vafer.jdeb.debian.ChangesFile;
30  import org.vafer.jdeb.utils.InformationOutputStream;
31  
32  /**
33   * Builds the Debian changes file.
34   */
35  class ChangesFileBuilder {
36  
37      private final Long outputTimestampMs;
38  
39      ChangesFileBuilder(Long outputTimestampMs) {
40          this.outputTimestampMs = outputTimestampMs;
41      }
42  
43      public ChangesFile createChanges(BinaryPackageControlFile packageControlFile, File binaryPackage, ChangesProvider changesProvider) throws IOException, PackagingException {
44  
45          ChangesFile changesFile = new ChangesFile();
46          changesFile.setChanges(changesProvider.getChangesSets());
47          changesFile.initialize(packageControlFile);
48  
49          changesFile.set("Date", outputTimestampMs == null ? ChangesFile.formatDate(new Date()) : ChangesFile.formatDateUTC(new Date(outputTimestampMs)));
50  
51          try {
52              // compute the checksums of the binary package
53              InformationOutputStream md5output = new InformationOutputStream(NullOutputStream.INSTANCE, MessageDigest.getInstance("MD5"));
54              InformationOutputStream sha1output = new InformationOutputStream(md5output, MessageDigest.getInstance("SHA1"));
55              InformationOutputStream sha256output = new InformationOutputStream(sha1output, MessageDigest.getInstance("SHA-256"));
56  
57              FileUtils.copyFile(binaryPackage, sha256output);
58  
59              // Checksums-Sha1:
60              //  56ef4c6249dc3567fd2967f809c42d1f9b61adf7 45964 jdeb.deb
61              changesFile.set("Checksums-Sha1", sha1output.getHexDigest() + " " + binaryPackage.length() + " " + binaryPackage.getName());
62  
63              // Checksums-Sha256:
64              //  38c6fa274eb9299a69b739bcbdbd05c7ffd1d8d6472f4245ed732a25c0e5d616 45964 jdeb.deb
65              changesFile.set("Checksums-Sha256", sha256output.getHexDigest() + " " + binaryPackage.length() + " " + binaryPackage.getName());
66  
67              StringBuilder files = new StringBuilder(md5output.getHexDigest());
68              files.append(' ').append(binaryPackage.length());
69              files.append(' ').append(packageControlFile.get("Section"));
70              files.append(' ').append(packageControlFile.get("Priority"));
71              files.append(' ').append(binaryPackage.getName());
72              changesFile.set("Files", files.toString());
73  
74          } catch (NoSuchAlgorithmException e) {
75              throw new PackagingException("Unable to compute the checksums for " + binaryPackage, e);
76          }
77  
78          if (!changesFile.isValid()) {
79              throw new PackagingException("Changes file fields are invalid " + changesFile.invalidFields() +
80                  ". The following fields are mandatory: " + changesFile.getMandatoryFields() +
81                  ". Please check your pom.xml/build.xml and your control file.");
82          }
83  
84          return changesFile;
85      }
86  }
