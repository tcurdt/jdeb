17  package org.vafer.jdeb.debian;
18  
19  import org.vafer.jdeb.utils.Utils;
20  import org.vafer.jdeb.changes.ChangeSet;
21  
22  import java.text.DateFormat;
23  import java.text.SimpleDateFormat;
24  import java.util.Date;
25  import java.util.Locale;
26  import java.util.Map.Entry;
27  import java.util.TimeZone;
28  
29  /**
30   * Reflecting a changes file
31   *
32   * @see <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html#s-debianchangesfiles">Debian Policy Manual - Debian changes files</a>
33   */
34  public final class ChangesFile extends ControlFile {
35  
36      private static final ControlField[] FIELDS = {
37          new ControlField("Format", true),
38          new ControlField("Date", true),
39          new ControlField("Source", true),
40          new ControlField("Binary", true),
41          new ControlField("Architecture", true),
42          new ControlField("Version", true),
43          new ControlField("Distribution", true),
44          new ControlField("Urgency", true),
45          new ControlField("Maintainer", true),
46          new ControlField("Changed-By"),
47          new ControlField("Description", true, ControlField.Type.MULTILINE, true),
48          new ControlField("Changes", true, ControlField.Type.MULTILINE, true),
49          new ControlField("Closes"),
50          new ControlField("Checksums-Sha1", true, ControlField.Type.MULTILINE, true),
51          new ControlField("Checksums-Sha256", true, ControlField.Type.MULTILINE, true),
52          new ControlField("Files", true, ControlField.Type.MULTILINE, true)
53      };
54  
55      public ChangesFile() {
56          set("Format", "1.8");
57          set("Urgency", "low");
58          set("Distribution", "stable");
59      }
60  
61      /**
62       * Initializes the fields on the changes file with the values of the specified
63       * binary package control file.
64       *
65       * @param packageControlFile
66       */
67      public void initialize(BinaryPackageControlFile packageControlFile) {
68          set("Binary",       packageControlFile.get("Package"));
69          set("Source",       Utils.defaultString(packageControlFile.get("Source"), packageControlFile.get("Package")));
70          set("Architecture", packageControlFile.get("Architecture"));
71          set("Version",      packageControlFile.get("Version"));
72          set("Maintainer",   packageControlFile.get("Maintainer"));
73          set("Distribution", packageControlFile.get("Distribution"));
74  
75          for (Entry<String, String> entry : packageControlFile.getUserDefinedFields().entrySet()) {
76              set(entry.getKey(), entry.getValue());
77          }
78  
79          StringBuilder description = new StringBuilder();
80          description.append(packageControlFile.get("Package"));
81          if (packageControlFile.get("Description") != null) {
82              description.append(" - ");
83              description.append(packageControlFile.getShortDescription());
84          }
85          set("Description",  description.toString());
86      }
87  
88      public void setChanges(ChangeSet[] changeSets) {
89          StringBuilder sb = new StringBuilder();
90  
91          if (changeSets.length > 0) {
92              final ChangeSet mostRecentChangeSet = changeSets[0];
93              set("Urgency", mostRecentChangeSet.getUrgency());
94              set("Changed-By", mostRecentChangeSet.getChangedBy());
95  
96              for (ChangeSet changeSet : changeSets) {
97                  sb.append(changeSet.toString());
98              }
99          }
100 
101         set("Changes", sb.toString());
102     }
103 
104     protected ControlField[] getFields() {
105         return FIELDS;
106     }
107 
108     protected char getUserDefinedFieldLetter() {
109         return 'C';
110     }
111 
112     public static String formatDate(Date date) {
113         final DateFormat format = createDateFormat();
114         return format.format(date);
115     }
116 
117     public static String formatDateUTC(Date date) {
118         final DateFormat format = createDateFormat();
119         format.setTimeZone(TimeZone.getTimeZone("UTC"));
120         return format.format(date);
121     }
122 
123     private static DateFormat createDateFormat() {
124         return new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH); // RFC 2822 format
125     }
126 }
