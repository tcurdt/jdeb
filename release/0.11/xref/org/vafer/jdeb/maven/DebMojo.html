259     protected VariableResolver initializeVariableResolver( Map variables ) {
260         variables.putAll(getProject().getProperties());
261 
262         variables.put("name", getProject().getName());
263         variables.put("artifactId", getProject().getArtifactId());
264         variables.put("groupId", getProject().getGroupId());
265         variables.put("version", getProjectVersion());
266         variables.put("description", getProject().getDescription());
267         variables.put("extension", "deb");
268         variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
269         variables.put("buildDir", buildDirectory.getAbsolutePath());
270         variables.put("project.version", getProject().getVersion());
271         variables.put("url", getProject().getUrl());
272 
273         return new MapVariableResolver(variables);

276     /**
277      * Doc some cleanup and conversion on the Maven project version.
278      * <ul>
279      * <li>any "-" is replaced by "+"</li>
280      * <li>"SNAPSHOT" is replaced with the current time and date, prepended by "~"</li>
281      * </ul>
282      *
283      * @return the Maven project version
284      */
285     private String getProjectVersion() {
286         String version = getProject().getVersion().replace('-', '+');
287 
288         if (this.timestamped && version.endsWith("+SNAPSHOT")) {
289             version = version.substring(0, version.length() - "+SNAPSHOT".length());
290             version += "~";
291             version += new SimpleDateFormat("yyyyMMdd.HHmmss.SSS").format(new Date());
292         }
293 
294         return version;
295     }

314     /**
315      * Main entry point
316      *
317      * @throws MojoExecutionException on error
318      */
319     public void execute() throws MojoExecutionException {
320 
321         final MavenProject project = getProject();
322 
323         if (isSubmodule() && !submodules) {
324             getLog().info("skipping sub module: jdeb executing at top-level only");
325             return;
326         }
327 
328         setData(dataSet);
329 
330         try {
331             Console infoConsole = new MojoConsole(getLog(), verbose);
332 
333             final VariableResolver resolver = initializeVariableResolver(new HashMap());
334 
335             final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
336             final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
337             final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
338             final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
339             final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
340             final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
341 
342             // if there are no producers defined we try to use the artifacts
343             if (dataProducers.isEmpty()) {
344 
345                 if (!hasMainArtifact()) {
346 
347                     final String packaging = project.getPackaging();
348                     if ("pom".equalsIgnoreCase(packaging)) {
349                         getLog().warn("Creating empty debian package.");
350                     } else {
351                         throw new MojoExecutionException(
352                             "Nothing to include into the debian package. " +
353                                 "Did you maybe forget to add a <data> tag or call the plugin directly?");
354                     }
355 
356                 } else {
357 
358                     Set<Artifact> artifacts = new HashSet<Artifact>();
359 
360                     artifacts.add(project.getArtifact());
361 
362                     for (Artifact artifact : (Set<Artifact>) project.getArtifacts()) {
363                         artifacts.add(artifact);
364                     }
365 
366                     for (Artifact artifact : (List<Artifact>) project.getAttachedArtifacts()) {
367                         artifacts.add(artifact);
368                     }
369 
370                     for(Artifact artifact : artifacts) {
371                         final File file = artifact.getFile();
372                         if (file != null) {
373                             dataProducers.add(new DataProducer() {
374                                 public void produce( final DataConsumer receiver ) {
375                                     try {
376                                         receiver.onEachFile(
377                                             new FileInputStream(file),
378                                             new File(installDirFile, file.getName()).getAbsolutePath(),
379                                             "",
380                                             "root", 0, "root", 0,
381                                             TarEntry.DEFAULT_FILE_MODE,
382                                             file.length());
383                                     } catch (Exception e) {
384                                         getLog().error(e);
385                                     }
386                                 }
387                             });
388                         } else {
389                             getLog().error("No file for artifact " + artifact);
390                         }
391                     }
392                 }
393             }
394 
395             try {
396 
397                 DebMaker debMaker = new DebMaker(infoConsole, debFile, controlDirFile, dataProducers, resolver);
398 
399                 if (changesInFile.exists() && changesInFile.canRead()) {
400                     debMaker.setChangesIn(changesInFile);
401                     debMaker.setChangesOut(changesOutFile);
402                     debMaker.setChangesSave(changesSaveFile);
403                 }
404 
405                 debMaker.setCompression(compression);
406                 debMaker.makeDeb();
407 
408                 // Always attach unless explicitly set to false
409                 if ("true".equalsIgnoreCase(attach)) {
410                     getLog().info("Attaching created debian archive " + debFile);
411                     projectHelper.attachArtifact(project, type, classifier, debFile);
412                 }
413 
414             } catch (PackagingException e) {
415                 getLog().error("Failed to create debian package " + debFile, e);
416                 throw new MojoExecutionException("Failed to create debian package " + debFile, e);
417             }
418 
419         } catch (ParseException e) {
420             throw new MojoExecutionException("Failed parsing pattern", e);
421         }
422     }
