095        private void addTo( final ArArchiveOutputStream pOutput, final String pName, final String pContent ) throws IOException {
096            final byte[] content = pContent.getBytes();
097            pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
098            pOutput.write(content);
099            pOutput.closeArchiveEntry();
100        }

102        private void addTo( final ArArchiveOutputStream pOutput, final String pName, final File pContent ) throws IOException {
103            pOutput.putArchiveEntry(new ArArchiveEntry(pName, pContent.length()));
104    
105            final InputStream input = new FileInputStream(pContent);
106            try {
107                Utils.copy(input, pOutput);
108            } finally {
109                input.close();
110            }
111    
112            pOutput.closeArchiveEntry();
113        }

115        /**
116         * Create the debian archive with from the provided control files and data producers.
117         *
118         * @param pControlFiles
119         * @param pData
120         * @param pOutput
121         * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
122         * @return PackageDescriptor
123         * @throws PackagingException
124         */
125        public PackageDescriptor createDeb( final File[] pControlFiles, final DataProducer[] pData, final File pOutput, String compression ) throws PackagingException {
126    
127            File tempData = null;
128            File tempControl = null;
129    
130            try {
131                tempData = File.createTempFile("deb", "data");
132                tempControl = File.createTempFile("deb", "control");
133    
134                console.info("Building data");
135                final StringBuilder md5s = new StringBuilder();
136                final BigInteger size = buildData(pData, tempData, md5s, compression);
137    
138                console.info("Building control");
139                final PackageDescriptor packageDescriptor = buildControl(pControlFiles, size, md5s, tempControl);
140    
141                if (!packageDescriptor.isValid()) {
142                    throw new PackagingException("Control file descriptor keys are invalid " + packageDescriptor.invalidKeys() +
143                        ". The following keys are mandatory " + Arrays.toString(PackageDescriptor.mandatoryKeys) +
144                        ". Please check your pom.xml/build.xml and your control file.");
145                }
146    
147                pOutput.getParentFile().mkdirs();
148    
149                // pass through stream chain to calculate all the different digests
150                final InformationOutputStream md5output = new InformationOutputStream(new FileOutputStream(pOutput), MessageDigest.getInstance("MD5"));
151                final InformationOutputStream sha1output = new InformationOutputStream(md5output, MessageDigest.getInstance("SHA1"));
152                final InformationOutputStream sha256output = new InformationOutputStream(sha1output, MessageDigest.getInstance("SHA-256"));
153                final ArArchiveOutputStream ar = new ArArchiveOutputStream(sha256output);
154    
155                addTo(ar, "debian-binary", "2.0\n");
156                addTo(ar, "control.tar.gz", tempControl);
157                addTo(ar, "data.tar" + getExtension(compression), tempData);
158    
159                ar.close();
160    
161                // intermediate values
162                packageDescriptor.set("MD5", md5output.getHexDigest());
163                packageDescriptor.set("SHA1", sha1output.getHexDigest());
164                packageDescriptor.set("SHA256", sha256output.getHexDigest());
165                packageDescriptor.set("Size", "" + md5output.getSize());
166                packageDescriptor.set("File", pOutput.getName());
167    
168                return packageDescriptor;
169    
170            } catch (Exception e) {
171                throw new PackagingException("Could not create deb package", e);
172            } finally {
173                if (tempData != null) {
174                    if (!tempData.delete()) {
175                        throw new PackagingException("Could not delete " + tempData);
176                    }
177                }
178                if (tempControl != null) {
179                    if (!tempControl.delete()) {
180                        throw new PackagingException("Could not delete " + tempControl);
181                    }
182                }
183            }
184        }

186        /**
187         * Return the extension of a file compressed with the specified method.
188         *
189         * @param pCompression the compression method used
190         * @return
191         */
192        private String getExtension( final String pCompression ) {
193            if ("gzip".equals(pCompression)) {
194                return ".gz";
195            } else if ("bzip2".equals(pCompression)) {
196                return ".bz2";
197            } else {
198                return "";
199            }

202        /**
203         * Create changes file based on the provided PackageDescriptor.
204         * If pRing, pKey and pPassphrase are provided the changes file will also be signed.
205         * It returns a ChangesDescriptor reflecting the changes
206         *
207         * @param pPackageDescriptor
208         * @param pChangesProvider
209         * @param pRing
210         * @param pKey
211         * @param pPassphrase
212         * @param pOutput
213         * @return ChangesDescriptor
214         * @throws IOException
215         * @throws PackagingException
216         */
217        public ChangesDescriptor createChanges( final PackageDescriptor pPackageDescriptor, final ChangesProvider pChangesProvider, final InputStream pRing, final String pKey, final String pPassphrase, final OutputStream pOutput ) throws IOException, PackagingException {
218    
219            final ChangeSet[] changeSets = pChangesProvider.getChangesSets();
220            final ChangesDescriptor changesDescriptor = new ChangesDescriptor(pPackageDescriptor, changeSets);
221    
222            changesDescriptor.set("Format", "1.8");
223    
224            if (changesDescriptor.get("Binary") == null) {
225                changesDescriptor.set("Binary", changesDescriptor.get("Package"));
226            }
227    
228            if (changesDescriptor.get("Source") == null) {
229                changesDescriptor.set("Source", changesDescriptor.get("Package"));
230            }
231    
232            final StringBuilder checksumsSha1 = new StringBuilder("\n");
233            // Checksums-Sha1:
234            // 56ef4c6249dc3567fd2967f809c42d1f9b61adf7 45964 jdeb.deb
235            checksumsSha1.append(' ').append(changesDescriptor.get("SHA1"));
236            checksumsSha1.append(' ').append(changesDescriptor.get("Size"));
237            checksumsSha1.append(' ').append(changesDescriptor.get("File"));
238            changesDescriptor.set("Checksums-Sha1", checksumsSha1.toString());
239    
240            final StringBuilder checksumsSha256 = new StringBuilder("\n");
241            // Checksums-Sha256:
242            // 38c6fa274eb9299a69b739bcbdbd05c7ffd1d8d6472f4245ed732a25c0e5d616 45964 jdeb.deb
243            checksumsSha256.append(' ').append(changesDescriptor.get("SHA256"));
244            checksumsSha256.append(' ').append(changesDescriptor.get("Size"));
245            checksumsSha256.append(' ').append(changesDescriptor.get("File"));
246            changesDescriptor.set("Checksums-Sha256", checksumsSha256.toString());
247    
248    
249            final StringBuilder files = new StringBuilder("\n");
250            files.append(' ').append(changesDescriptor.get("MD5"));
251            files.append(' ').append(changesDescriptor.get("Size"));
252            files.append(' ').append(changesDescriptor.get("Section"));
253            files.append(' ').append(changesDescriptor.get("Priority"));
254            files.append(' ').append(changesDescriptor.get("File"));
255            changesDescriptor.set("Files", files.toString());
256    
257            if (!changesDescriptor.isValid()) {
258                throw new PackagingException("Changes file descriptor keys are invalid " + changesDescriptor.invalidKeys() +
259                    ". The following keys are mandatory " + Arrays.toString(ChangesDescriptor.mandatoryKeys) +
260                    ". Please check your pom.xml/build.xml and your control file.");
261            }
262    
263            final String changes = changesDescriptor.toString();
264            final byte[] changesBytes = changes.getBytes("UTF-8");
265    
266            if (pRing == null || pKey == null || pPassphrase == null) {
267                pOutput.write(changesBytes);
268                pOutput.close();
269                return changesDescriptor;
270            }
271    
272            console.info("Signing changes with key " + pKey);
273    
274            final InputStream input = new ByteArrayInputStream(changesBytes);
275    
276            try {
277                SigningUtils.clearSign(input, pRing, pKey, pPassphrase, pOutput);
278            } catch (Exception e) {
279                e.printStackTrace();
280            }
281    
282            pOutput.close();
283    
284            return changesDescriptor;
285        }

287        /**
288         * Build control archive of the deb
289         *
290         * @param pControlFiles
291         * @param pDataSize
292         * @param pChecksums
293         * @param pOutput
294         * @return
295         * @throws FileNotFoundException
296         * @throws IOException
297         * @throws ParseException
298         */
299        private PackageDescriptor buildControl( final File[] pControlFiles, final BigInteger pDataSize, final StringBuilder pChecksums, final File pOutput ) throws IOException, ParseException {
300    
301            final File dir = pOutput.getParentFile();
302            if (dir != null && (!dir.exists() || !dir.isDirectory())) {
303                throw new IOException("Cannot write control file at '" + pOutput.getAbsolutePath() + "'");
304            }
305    
306            final TarOutputStream outputStream = new TarOutputStream(new GZIPOutputStream(new FileOutputStream(pOutput)));
307            outputStream.setLongFileMode(TarOutputStream.LONGFILE_GNU);
308    
309    
310            // create a descriptor out of the "control" file, copy all other files, ignore directories
311            PackageDescriptor packageDescriptor = null;
312            for (File file : pControlFiles) {
313                if (file.isDirectory()) {
314                    // warn about the misplaced directory, except for directories ignored by default (.svn, cvs, etc)
315                    boolean isDefaultExcludes = false;
316                    for (String pattern : DirectoryScanner.getDefaultExcludes()) {
317                        isDefaultExcludes = DirectoryScanner.match(pattern, file.getAbsolutePath().replace("\\", "/"));
318                        if (isDefaultExcludes) {
319                            break;
320                        }
321                    }
322    
323                    if (!isDefaultExcludes) {
324                        console.info("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
325                    }
326                    continue;
327                }
328    
329                final TarEntry entry = new TarEntry(file);
330                final String name = file.getName();
331    
332                entry.setName("./" + name);
333                entry.setNames("root", "root");
334                entry.setMode(PermMapper.toMode("755"));
335    
336                if (CONFIGURATION_FILENAMES.contains(name)) {
337    
338                    FilteredConfigurationFile configurationFile = new FilteredConfigurationFile(file.getName(), new FileInputStream(file), resolver);
339                    configurationFiles.add(configurationFile);
340    
341                } else if ("control".equals(name)) {
342    
343                    packageDescriptor = new PackageDescriptor(new FileInputStream(file), resolver);
344    
345                    if (packageDescriptor.get("Date") == null) {
346                        // Mon, 26 Mar 2007 11:44:04 +0200 (RFC 2822)
347                        SimpleDateFormat fmt = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH);
348                        packageDescriptor.set("Date", fmt.format(new Date()));
349                    }
350    
351                    if (packageDescriptor.get("Distribution") == null) {
352                        packageDescriptor.set("Distribution", "unknown");
353                    }
354    
355                    if (packageDescriptor.get("Urgency") == null) {
356                        packageDescriptor.set("Urgency", "low");
357                    }
358    
359                    packageDescriptor.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
360    
361                    final String debFullName = System.getenv("DEBFULLNAME");
362                    final String debEmail = System.getenv("DEBEMAIL");
363    
364                    if (debFullName != null && debEmail != null) {
365                        final String maintainer = debFullName + " <" + debEmail + ">";
366                        packageDescriptor.set("Maintainer", maintainer);
367                        console.info("Using maintainer '" + maintainer + "' from the environment variables.");
368                    }
369    
370                } else {
371    
372                    InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
373                    Utils.copy(infoStream, NullOutputStream.NULL_OUTPUT_STREAM);
374                    infoStream.close();
375    
376                    InputStream in = new FileInputStream(file);
377                    if (infoStream.isShell() && !infoStream.hasUnixLineEndings()) {
378                        // fix the line endings automatically
379                        byte[] buf = Utils.toUnixLineEndings(in);
380                        entry.setSize(buf.length);
381                        in = new ByteArrayInputStream(buf);
382                    }
383    
384                    outputStream.putNextEntry(entry);
385                    Utils.copy(in, outputStream);
386                    outputStream.closeEntry();
387                    in.close();
388                }
389            }
390    
391            if (packageDescriptor == null) {
392                throw new FileNotFoundException("No 'control' found in " + Arrays.toString(pControlFiles));
393            }
394    
395            for (FilteredConfigurationFile configurationFile : configurationFiles) {
396                addControlEntry(configurationFile.getName(), configurationFile.toString(), outputStream);
397            }
398            addEntry("control", packageDescriptor.toString(), outputStream);
399            addEntry("md5sums", pChecksums.toString(), outputStream);
400    
401            outputStream.close();
402    
403            return packageDescriptor;
404        }

407        // FIXME temporary - only until Commons Compress is fixed
408        private OutputStream compressedOutputStream( String pCompression, final OutputStream outputStream ) throws CompressorException {
409            if ("none".equalsIgnoreCase(pCompression)) {
410                return outputStream;
411            }
412            if ("gzip".equals(pCompression)) {
413                pCompression = "gz";
414            }
415            return new CompressorStreamFactory().createCompressorOutputStream(pCompression, outputStream);
416        }

418        /**
419         * Build the data archive of the deb from the provided DataProducers
420         *
421         * @param pData
422         * @param pOutput
423         * @param pChecksums
424         * @param pCompression the compression method used for the data file (gzip, bzip2 or anything else for no compression)
425         * @return
426         * @throws NoSuchAlgorithmException
427         * @throws IOException
428         * @throws CompressorException
429         */
430        BigInteger buildData( final DataProducer[] pData, final File pOutput, final StringBuilder pChecksums, String pCompression ) throws NoSuchAlgorithmException, IOException, CompressorException {
431    
432            final File dir = pOutput.getParentFile();
433            if (dir != null && (!dir.exists() || !dir.isDirectory())) {
434                throw new IOException("Cannot write data file at '" + pOutput.getAbsolutePath() + "'");
435            }
436    
437            final TarOutputStream tarOutputStream = new TarOutputStream(compressedOutputStream(pCompression, new FileOutputStream(pOutput)));
438            tarOutputStream.setLongFileMode(TarOutputStream.LONGFILE_GNU);
439    
440            final MessageDigest digest = MessageDigest.getInstance("MD5");
441    
442            final Total dataSize = new Total();
443    
444            final List addedDirectories = new ArrayList();
445            final DataConsumer receiver = new DataConsumer() {
446                public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
447                    dirname = fixPath(dirname);
448    
449                    createParentDirectories((new File(dirname)).getParent(), user, uid, group, gid);
450    
451                    // The directory passed in explicitly by the caller also gets the passed-in mode.  (Unlike
452                    // the parent directories for now.  See related comments at "int mode =" in
453                    // createParentDirectories, including about a possible bug.)
454                    createDirectory(dirname, user, uid, group, gid, mode, 0);
455    
456                    console.info("dir: " + dirname);
457                }
458    
459                public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
460                    filename = fixPath(filename);
461    
462                    createParentDirectories((new File(filename)).getParent(), user, uid, group, gid);
463    
464                    final TarEntry entry = new TarEntry(filename);
465    
466                    // FIXME: link is in the constructor
467                    entry.setUserName(user);
468                    entry.setUserId(uid);
469                    entry.setGroupName(group);
470                    entry.setGroupId(gid);
471                    entry.setMode(mode);
472                    entry.setSize(size);
473    
474                    tarOutputStream.putNextEntry(entry);
475    
476                    dataSize.add(size);
477                    digest.reset();
478    
479                    Utils.copy(inputStream, new DigestOutputStream(tarOutputStream, digest));
480    
481                    final String md5 = Utils.toHex(digest.digest());
482    
483                    tarOutputStream.closeEntry();
484    
485                    console.info(
486                        "file:" + entry.getName() +
487                            " size:" + entry.getSize() +
488                            " mode:" + entry.getMode() +
489                            " linkname:" + entry.getLinkName() +
490                            " username:" + entry.getUserName() +
491                            " userid:" + entry.getUserId() +
492                            " groupname:" + entry.getGroupName() +
493                            " groupid:" + entry.getGroupId() +
494                            " modtime:" + entry.getModTime() +
495                            " md5: " + md5
496                    );
497    
498                    // append to file md5 list
499                    pChecksums.append(md5).append(" ").append(entry.getName()).append('\n');
500                }

502                private String fixPath( String path ) {
503                    // If we're receiving directory names from Windows, then we'll convert to use slash
504                    // This does eliminate the ability to use of a backslash in a directory name on *NIX,
505                    // but in practice, this is a non-issue
506                    if (path.indexOf('\\') > -1) {
507                        path = path.replace('\\', '/');
508                    }
509                    // ensure the path is like : ./foo/bar
510                    if (path.startsWith("/")) {
511                        path = "." + path;
512                    } else if (!path.startsWith("./")) {
513                        path = "./" + path;
514                    }
515                    return path;
516                }

518                private void createDirectory( String directory, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
519                    // All dirs should end with "/" when created, or the test DebAndTaskTestCase.testTarFileSet() thinks its a file
520                    // and so thinks it has the wrong permission.
521                    // This consistency also helps when checking if a directory already exists in addedDirectories.
522    
523                    if (!directory.endsWith("/")) {
524                        directory += "/";
525                    }
526    
527                    if (!addedDirectories.contains(directory)) {
528                        TarEntry entry = new TarEntry(directory);
529                        entry.setUserName(user);
530                        entry.setUserId(uid);
531                        entry.setGroupName(group);
532                        entry.setGroupId(gid);
533                        entry.setMode(mode);
534                        entry.setSize(size);
535    
536                        tarOutputStream.putNextEntry(entry);
537                        tarOutputStream.closeEntry();
538                        addedDirectories.add(directory); // so addedDirectories consistently have "/" for finding duplicates.
539                    }
540                }
541    
542                private void createParentDirectories( String dirname, String user, int uid, String group, int gid ) throws IOException {
543                    // Debian packages must have parent directories created
544                    // before sub-directories or files can be installed.
545                    // For example, if an entry of ./usr/lib/foo/bar existed
546                    // in a .deb package, but the ./usr/lib/foo directory didn't
547                    // exist, the package installation would fail.  The .deb must
548                    // then have an entry for ./usr/lib/foo and then ./usr/lib/foo/bar
549    
550                    if (dirname == null) {
551                        return;
552                    }
553    
554                    // The loop below will create entries for all parent directories
555                    // to ensure that .deb packages will install correctly.
556                    String[] pathParts = dirname.split("\\/");
557                    String parentDir = "./";
558                    for (int i = 1; i < pathParts.length; i++) {
559                        parentDir += pathParts[i] + "/";
560                        // Make it so the dirs can be traversed by users.
561                        // We could instead try something more granular, like setting the directory
562                        // permission to 'rx' for each of the 3 user/group/other read permissions
563                        // found on the file being added (ie, only if "other" has read
564                        // permission on the main node, then add o+rx permission on all the containing
565                        // directories, same w/ user & group), and then also we'd have to
566                        // check the parentDirs collection of those already added to
567                        // see if those permissions need to be similarly updated.  (Note, it hasn't
568                        // been demonstrated, but there might be a bug if a user specifically
569                        // requests a directory with certain permissions,
570                        // that has already been auto-created because it was a parent, and if so, go set
571                        // the user-requested mode on that directory instead of this automatic one.)
572                        // But for now, keeping it simple by making every dir a+rx.   Examples are:
573                        // drw-r----- fs/fs   # what you get with setMode(mode)
574                        // drwxr-xr-x fs/fs   # Usable. Too loose?
575                        int mode = TarEntry.DEFAULT_DIR_MODE;
576    
577                        createDirectory(parentDir, user, uid, group, gid, mode, 0);
578                    }
579                }
580            };
581    
582            try {
583                for (int i = 0; i < pData.length; i++) {
584                    final DataProducer data = pData[i];
585                    data.produce(receiver);
586                }
587            } finally {
588                tarOutputStream.close();
589            }
590    
591            console.info("Total size: " + dataSize);
592    
593            return dataSize.count;
594        }

596        private static void addEntry( final String pName, final String pContent, final TarOutputStream pOutput ) throws IOException {
597            final byte[] data = pContent.getBytes("UTF-8");
598    
599            final TarEntry entry = new TarEntry("./" + pName);
600            entry.setSize(data.length);
601            entry.setNames("root", "root");
602    
603            pOutput.putNextEntry(entry);
604            pOutput.write(data);
605            pOutput.closeEntry();
606        }

612            entry.setSize(data.length);
