046    public static int copy( final InputStream pInput, final OutputStream pOutput ) throws IOException {
047        final byte[] buffer = new byte[2048];
048        int count = 0;
049        int n;
050        while (-1 != (n = pInput.read(buffer))) {
051            pOutput.write(buffer, 0, n);
052            count += n;
053        }
054        return count;
055    }

057    public static String toHex( final byte[] bytes ) {
058        final StringBuilder sb = new StringBuilder();
059
060        for (byte b : bytes) {
061            sb.append(Integer.toHexString((b >> 4) & 0x0f));
062            sb.append(Integer.toHexString(b & 0x0f));
063        }
064
065        return sb.toString();
066    }

068    public static String stripPath( final int p, final String s ) {
069
070        if (p <= 0) {
071            return s;
072        }
073
074        int x = 0;
075        for (int i = 0; i < p; i++) {
076            x = s.indexOf('/', x + 1);
077            if (x < 0) {
078                return s;
079            }
080        }
081
082        return s.substring(x + 1);
083    }

085    public static String stripLeadingSlash( final String s ) {
086        if (s == null) {
087            return s;
088        }
089        if (s.length() == 0) {
090            return s;
091        }
092        if (s.charAt(0) == '/' || s.charAt(0) == '\\') {
093            return s.substring(1);
094        }
095        return s;
096    }

098    /**
099     * Substitute the variables in the given expression with the
100     * values from the resolver
101     *
102     * @param pResolver
103     * @param pExpression
104     */
105    public static String replaceVariables( final VariableResolver pResolver, final String pExpression, final String pOpen, final String pClose ) {
106        final char[] open = pOpen.toCharArray();
107        final char[] close = pClose.toCharArray();
108        
109        final StringBuilder out = new StringBuilder();
110        StringBuilder sb = new StringBuilder();
111        char[] last = null;
112        int wo = 0;
113        int wc = 0;
114        int level = 0;
115        for (char c : pExpression.toCharArray()) {
116            if (c == open[wo]) {
117                wo++;
118                if (open.length == wo) {
119                    // found open
120                    if (last == open) {
121                        out.append(open);
122                    }
123                    level++;
124                    out.append(sb);
125                    sb = new StringBuilder();
126                    wo = 0;
127                    last = open;
128                }
129            } else if (c == close[wc]) {
130                wc++;
131                if (close.length == wc) {
132                    // found close
133                    if (last == open) {
134                        final String variable = pResolver.get(sb.toString());
135                        if (variable != null) {
136                            out.append(variable);
137                        } else {
138                            out.append(open);
139                            out.append(sb);
140                            out.append(close);
141                        }
142                    } else {
143                        out.append(sb);
144                        out.append(close);
145                    }
146                    sb = new StringBuilder();
147                    level--;
148                    wc = 0;
149                    last = close;
150                }
151            } else {
152
153                if (wo > 0) {
154                    sb.append(open, 0, wo);
155                }
156
157                if (wc > 0) {
158                    sb.append(close, 0, wc);
159                }
160
161                sb.append(c);
162
163                wo = wc = 0;
164            }
165        }
166
167        if (wo > 0) {
168            sb.append(open, 0, wo);
169        }
170
171        if (wc > 0) {
172            sb.append(close, 0, wc);
173        }
174
175        if (level > 0) {
176            out.append(open);
177        }
178        out.append(sb);
179
180        return out.toString();
181    }

183    /**
184     * Replaces new line delimiters in the input stream with the Unix line feed.
185     *
186     * @param input
187     */
188    public static byte[] toUnixLineEndings( InputStream input ) throws IOException {
189        String encoding = "ISO-8859-1";
190        FixCrLfFilter filter = new FixCrLfFilter(new InputStreamReader(input, encoding));
191        filter.setEol(FixCrLfFilter.CrLf.newInstance("unix"));
192
193        ByteArrayOutputStream filteredFile = new ByteArrayOutputStream();
194        Utils.copy(new ReaderInputStream(filter, encoding), filteredFile);
195
196        return filteredFile.toByteArray();
197    }

199    /**
200     * Convert the project version to a version suitable for a Debian package.
201     * -SNAPSHOT suffixes are replaced with a timestamp (~yyyyMMddHHmmss).
202     * The separator before a rc, alpha or beta version is replaced with '~'
203     * such that the version is always ordered before the final or GA release.
204     * 
205     * @param version the project version to convert to a Debian package version
206     * @param timestamp the date used as the timestamp to replace the SNAPSHOT suffix
207     */
208    public static String convertToDebianVersion( String version, boolean apply, String envName, Date timestamp ) {
209        Pattern pattern1 = Pattern.compile("(.*)[\\-\\+]SNAPSHOT");
210        Matcher matcher = pattern1.matcher(version);
211        if (matcher.matches()) {
212            version = matcher.group(1) + "~";
213
214            if (apply) {
215                final String envValue = System.getenv(envName);
216                final String snapshot = (envValue != null && envValue.length() > 0)
217                        ? envValue
218                        : new SimpleDateFormat("yyyyMMddHHmmss").format(timestamp);
219                version += snapshot;
220            } else {
221                version += "SNAPSHOT";
222            }
223        }
224        
225        Pattern pattern2 = Pattern.compile("(.*?)([\\.\\-_]?)(alpha|beta|rc)(.*)", Pattern.CASE_INSENSITIVE);
226        matcher = pattern2.matcher(version);
227        if (matcher.matches()) {
228            version = matcher.group(1) + "~" + matcher.group(3) + matcher.group(4);
229        }
230        
231        version = version.replace('-', '+');
232        
233        return version;
234    }

236    /**
237     * Construct new path by replacing file directory part. No
238     * files are actually modified.
239     * @param file path to move
240     * @param target new path directory
241     */
242    public static String movePath( final String file,
243                                   final String target ) {
244        final String name = new File(file).getName();
245        return target.endsWith("/") ? target + name : target + '/' + name;
246    }

261    /**
262     * Get the known locations where the secure keyring can be located.
263     * Looks through known locations of the GNU PG secure keyring.
264     * 
265     * @return The location of the PGP secure keyring if it was found,
266     *         null otherwise
267     */
268    public static Collection<String> getKnownPGPSecureRingLocations() {
269        final LinkedHashSet<String> locations = new LinkedHashSet<String>();
270
271        final String os = System.getProperty("os.name");
272        final boolean runOnWindows = os == null || os.toLowerCase().contains("win");
273
274        if (runOnWindows) {
275            // The user's roaming profile on Windows, via environment
276            final String windowsRoaming = System.getenv("APPDATA");
277            if (windowsRoaming != null) {
278                locations.add(joinPaths(windowsRoaming, "gnupg", "secring.gpg"));
279            }
280
281            // The user's local profile on Windows, via environment
282            final String windowsLocal = System.getenv("LOCALAPPDATA");
283            if (windowsLocal != null) {
284                locations.add(joinPaths(windowsLocal, "gnupg", "secring.gpg"));
285            }
286
287            // The Windows installation directory
288            final String windir = System.getProperty("WINDIR");
289            if (windir != null) {
290                // Local Profile on Windows 98 and ME
291                locations.add(joinPaths(windir, "Application Data", "gnupg", "secring.gpg"));
292            }
293        }
294
295        final String home = System.getProperty("user.home");
296
297        if (home != null && runOnWindows) {
298            // These are for various flavours of Windows
299            // if the environment variables above have failed
300
301            // Roaming profile on Vista and later
302            locations.add(joinPaths(home, "AppData", "Roaming", "gnupg", "secring.gpg"));
303            // Local profile on Vista and later
304            locations.add(joinPaths(home, "AppData", "Local", "gnupg", "secring.gpg"));
305            // Roaming profile on 2000 and XP
306            locations.add(joinPaths(home, "Application Data", "gnupg", "secring.gpg"));
307            // Local profile on 2000 and XP
308            locations.add(joinPaths(home, "Local Settings", "Application Data", "gnupg", "secring.gpg"));
309        }
310
311        // *nix, including OS X
312        if (home != null) {
313            locations.add(joinPaths(home, ".gnupg", "secring.gpg"));
314        }
315
316        return locations;
317    }

319    /**
320     * Tries to guess location of the user secure keyring using various
321     * heuristics.
322     *
323     * @return path to the keyring file
324     * @throws FileNotFoundException if no keyring file found
325     */
326    public static File guessKeyRingFile() throws FileNotFoundException {
327        final Collection<String> possibleLocations = getKnownPGPSecureRingLocations();
328        for (final String location : possibleLocations) {
329            final File candidate = new File(location);
330            if (candidate.exists()) {
331                return candidate;
332            }
333        }
334        final StringBuilder message = new StringBuilder("Could not locate secure keyring, locations tried: ");
335        final Iterator<String> it = possibleLocations.iterator();
336        while (it.hasNext()) {
337            message.append(it.next());
338            if (it.hasNext()) {
339                message.append(", ");
340            }
341        }
342        throw new FileNotFoundException(message.toString());
343    }

345    /**
346     * Join together path elements with File.separator. Filters out null
347     * elements.
348     * 
349     * @param elements The path elements to join
350     * @return elements concatenated together with File.separator
351     */
352    public static String joinPaths(String... elements) {
353        StringBuilder builder = new StringBuilder();
354        boolean first = true;
355        for (String element : elements) {
356            // Skip null elements
357            if (element == null) {
358                // This won't change the value of first if we skip elements
359                // in the beginning of the array
360                continue;
361            }
362            if (!first) {
363                builder.append(File.separatorChar);
364            }
365            builder.append(element);
366            first = false;
367        }
368        return builder.toString();
369    }
