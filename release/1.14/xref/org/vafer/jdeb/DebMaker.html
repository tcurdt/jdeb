300     public void validate() throws PackagingException {
301         if (control == null || !control.isDirectory()) {
302             throw new PackagingException("The 'control' attribute doesn't point to a directory. " + control);
303         }
304 
305         if (changesEnabled) {
306             if (changesIn != null) {
307 
308                 if (changesIn.exists() && (!changesIn.isFile() || !changesIn.canRead())) {
309                     throw new PackagingException("The 'changesIn' setting needs to point to a readable file. " + changesIn + " was not found/readable.");
310                 }
311 
312                 if (changesOut != null && !isWritableFile(changesOut)) {
313                     throw new PackagingException("Cannot write the output for 'changesOut' to " + changesOut);
314                 }
315 
316                 if (changesSave != null && !isWritableFile(changesSave)) {
317                     throw new PackagingException("Cannot write the output for 'changesSave' to " + changesSave);
318                 }
319 
320             } else {
321                 if (changesOut != null || changesSave != null) {
322                     throw new PackagingException("The 'changesOut' or 'changesSave' settings may only be used when there is a 'changesIn' specified.");
323                 }
324             }
325         }
326 
327         if (Compression.toEnum(compression) == null) {
328             throw new PackagingException("The compression method '" + compression + "' is not supported (expected 'none', 'gzip', 'bzip2' or 'xz')");
329         }
330 
331         if (deb == null) {
332             throw new PackagingException("You need to specify where the deb file is supposed to be created.");
333         }
334 
335         PGPSigner.getDigestCode(digest);
336     }

338     public void makeDeb() throws PackagingException {
339         BinaryPackageControlFile packageControlFile;
340         try {
341             console.info("Creating debian package: " + deb);
342 
343             // If we should sign the package
344             if (signPackage) {
345 
346                 if (keyring == null || !keyring.exists()) {
347                     console.warn("Signing requested, but no keyring supplied");
348                 }
349 
350                 if (key == null) {
351                     console.warn("Signing requested, but no key supplied");
352                 }
353 
354                 if (passphrase == null) {
355                     console.warn("Signing requested, but no passphrase supplied");
356                 }
357 
358                 final int digestCode = PGPSigner.getDigestCode(signDigest);
359 
360                 PGPSigner signer;
361                 try (FileInputStream keyRingInput = new FileInputStream(keyring)) {
362                     signer = new PGPSigner(keyRingInput, key, passphrase, digestCode);
363                 }
364 
365                 PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(signer.getSecretKey().getPublicKey().getAlgorithm(), digestCode));
366                 signatureGenerator.init(PGPSignature.BINARY_DOCUMENT, signer.getPrivateKey());
367 
368                 packageControlFile = createSignedDeb(Compression.toEnum(compression), signatureGenerator, signer);
369             } else {
370                 packageControlFile = createDeb(Compression.toEnum(compression));
371             }
372 
373         } catch (Exception e) {
374             throw new PackagingException("Failed to create debian package " + deb, e);
375         }
376 
377         if(changesEnabled)
378         {
379             makeChangesFiles(packageControlFile);
380         }
381     }

383     private void makeChangesFiles(final BinaryPackageControlFile packageControlFile) throws PackagingException {
384         if (changesOut == null) {
385             changesOut = new File(deb.getParentFile(), FilenameUtils.getBaseName(deb.getName()) + ".changes");
386         }
387 
388         ChangesProvider changesProvider;
389         FileOutputStream out = null;
390 
391         try {
392             console.info("Creating changes file: " + changesOut);
393 
394             changesOut.getParentFile().mkdirs();
395 
396             out = new FileOutputStream(changesOut);
397 
398             if (changesIn != null && changesIn.exists()) {
399                 // read the changes form a textfile provider
400                 changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageControlFile, outputTimestampMs, encoding);
401             } else {
402                 // create an empty changelog
403                 changesProvider = () -> new ChangeSet[] {
404                         new ChangeSet(packageControlFile.get("Package"),
405                                 packageControlFile.get("Version"),
406                                 outputTimestampMs == null ? new Date() : new Date(outputTimestampMs),
407                                 packageControlFile.get("Distribution"),
408                                 packageControlFile.get("Urgency"),
409                                 packageControlFile.get("Maintainer"),
410                                 new String[0])
411                 };
412             }
413 
414             ChangesFileBuilder builder = new ChangesFileBuilder(outputTimestampMs);
415             ChangesFile changesFile = builder.createChanges(packageControlFile, deb, changesProvider);
416 
417             final int digestCode = PGPSigner.getDigestCode(signDigest);
418 
419             // (signChanges || signPackage) - for backward compatibility. signPackage is signing both changes and deb.
420             if ((signChanges || signPackage) && keyring != null && key != null && passphrase != null) {
421                 console.info("Signing the changes file with the key " + key);
422                 PGPSigner signer = new PGPSigner(new FileInputStream(keyring), key, passphrase, digestCode);
423                 signer.clearSign(changesFile.toString(), out);
424             } else {
425                 out.write(changesFile.toString().getBytes(StandardCharsets.UTF_8));
426             }
427             out.flush();
428 
429         } catch (Exception e) {
430             throw new PackagingException("Failed to create the Debian changes file " + changesOut, e);
431         } finally {
432             IOUtils.closeQuietly(out);
433         }
434 
435         if (changesSave == null || !(changesProvider instanceof TextfileChangesProvider)) {
436             return;
437         }
438 
439         try {
440             console.info("Saving changes to file: " + changesSave);
441 
442             ((TextfileChangesProvider) changesProvider).save(new FileOutputStream(changesSave));
443 
444         } catch (Exception e) {
445             throw new PackagingException("Failed to save debian changes file " + changesSave, e);
446         }
447     }

449     private List<String> populateConffiles(Collection<DataProducer> producers) {
450         final List<String> result = new ArrayList<>();
451 
452         if (producers == null || producers.isEmpty()) {
453             return result;
454         }
455 
456         final DataConsumer receiver = new DataConsumer() {
457             public void onEachFile(InputStream input, TarArchiveEntry entry)  {
458                 String tempConffileItem = entry.getName();
459 
460                 // Make sure the conffile path is absolute
461                 if (tempConffileItem.startsWith(".")) {
462                     tempConffileItem = tempConffileItem.substring(1);
463                 }
464                 if (!tempConffileItem.startsWith("/")) {
465                     tempConffileItem = "/" + tempConffileItem;
466                 }
467 
468                 console.info("Adding conffile: " + tempConffileItem);
469                 result.add(tempConffileItem);
470             }
471 
472             public void onEachLink(TarArchiveEntry entry)  {
473             }
474 
475             public void onEachDir(TarArchiveEntry tarArchiveEntry)  {
476             }
477         };
478 
479         try {
480             for (DataProducer data : producers) {
481                 data.produce(receiver);
482             }
483         } catch(Exception e) {
484             //
485         }
486 
487         return result;
488     }

500     /**
501      * Create the debian archive with from the provided control files and data producers.
502      *
503      * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
504      * @param signatureGenerator   the signature generator
505      *
506      * @return PackageDescriptor
507      * @throws PackagingException
508      */
509     public BinaryPackageControlFile createSignedDeb(Compression compression, final PGPSignatureGenerator signatureGenerator, PGPSigner signer ) throws PackagingException {
510         File tempData = null;
511         File tempControl = null;
512 
513         try {
514             tempData = File.createTempFile("deb", "data");
515             tempControl = File.createTempFile("deb", "control");
516 
517             console.debug("Building data");
518             DataBuilder dataBuilder = new DataBuilder(console, outputTimestampMs);
519             StringBuilder md5s = new StringBuilder();
520             TarOptions options = new TarOptions()
521                 .compression(compression)
522                 .longFileMode(tarLongFileMode)
523                 .bigNumberMode(tarBigNumberMode);
524             BigInteger size = dataBuilder.buildData(dataProducers, tempData, md5s, options, ignoreBrokenLinks);
525 
526             console.info("Building conffiles");
527             List<String> tempConffiles = populateConffiles(conffilesProducers);
528 
529             console.debug("Building control");
530             ControlBuilder controlBuilder = new ControlBuilder(console, variableResolver, openReplaceToken, closeReplaceToken, outputTimestampMs);
531             BinaryPackageControlFile packageControlFile = controlBuilder.createPackageControlFile(new File(control, "control"), size, encoding, openReplaceToken, closeReplaceToken);
532             if (packageControlFile.get("Package") == null) {
533                 packageControlFile.set("Package", packageName);
534             }
535             if (packageControlFile.get("Section") == null) {
536                 packageControlFile.set("Section", section);
537             }
538             if (packageControlFile.get("Description") == null) {
539                 packageControlFile.set("Description", description);
540             }
541             if (packageControlFile.get("Depends") == null) {
542                 // Only add a depends entry to the control file if the field in this object has actually been set
543                 if (depends != null && depends.length() > 0) {
544                     packageControlFile.set("Depends", depends);
545                 }
546             }
547             if (packageControlFile.get("Homepage") == null) {
548                 packageControlFile.set("Homepage", homepage);
549             }
550 
551             controlBuilder.buildControl(packageControlFile, control.listFiles(), tempConffiles , md5s, tempControl, encoding);
552 
553             if (!packageControlFile.isValid()) {
554                 throw new PackagingException("Control file fields are invalid " + packageControlFile.invalidFields() +
555                         ". The following fields are mandatory: " + packageControlFile.getMandatoryFields() +
556                         ". Please check your pom.xml/build.xml and your control file.");
557             }
558 
559             deb.getParentFile().mkdirs();
560 
561             ArArchiveOutputStream ar = new ArArchiveOutputStream(new FileOutputStream(deb));
562 
563             String binaryName = "debian-binary";
564             String binaryContent = "2.0\n";
565             String controlName = "control.tar.gz";
566             String dataName = "data.tar" + compression.getExtension();
567 
568             addTo(ar, binaryName, binaryContent);
569             addTo(ar, controlName, tempControl);
570             addTo(ar, dataName, tempData);
571 
572             if (signatureGenerator != null) {
573                 console.info("Signing package with key " + key);
574 
575                 if(signRole == null) {
576                     signRole = "origin";
577                 }
578 
579                 // Use debsig-verify as default
580                 if (!"dpkg-sig".equals(signMethod)) {
581                     // Sign file to verify with debsig-verify
582                     PGPSignatureOutputStream sigStream = new PGPSignatureOutputStream(signatureGenerator);
583 
584                     addTo(sigStream, binaryContent);
585                     addTo(sigStream, tempControl);
586                     addTo(sigStream, tempData);
587                     addTo(ar, "_gpg" + signRole, sigStream.generateASCIISignature());
588 
589                 } else {
590 
591                     // Sign file to verify with dpkg-sig --verify
592                     final String outputStr =
593                             "Version: 4\n" +
594                                     "Signer: \n" +
595                                     "Date: " + new SimpleDateFormat("EEE MMM dd HH:mm:ss yyyy", Locale.ENGLISH).format(new Date()) + "\n" +
596                                     "Role: " + signRole +"\n" +
597                                     "Files: \n" +
598                                     addFile(binaryName, binaryContent) +
599                                     addFile(controlName, tempControl) +
600                                     addFile(dataName, tempData);
601 
602                     ByteArrayOutputStream message = new ByteArrayOutputStream();
603                     signer.clearSign(outputStr, message);
604 
605                     addTo(ar, "_gpg" + signRole, message.toString());
606                 }
607             }
608 
609             ar.close();
610 
611             return packageControlFile;
612 
613         } catch (Exception e) {
614             throw new PackagingException("Could not create deb package", e);
615         } finally {
616             if (tempData != null) {
617                 if (!tempData.delete()) {
618                     console.warn("Could not delete the temporary file " + tempData);
619                 }
620             }
621             if (tempControl != null) {
622                 if (!tempControl.delete()) {
623                     console.warn("Could not delete the temporary file " + tempControl);
624                 }
625             }
626         }
627     }

656     private String md5Hash(byte[] input){
657         //update the input of MD5
658         MD5Digest md5 = new MD5Digest();
659         md5.update(input, 0, input.length);
660 
661         //get the output/ digest size and hash it
662         byte[] digest = new byte[md5.getDigestSize()];
663         md5.doFinal(digest, 0);
664 
665         return new String(Hex.encode(digest));
666     }

683     private String sha1Hash(byte[] input){
684         try
685         {
686             //prepare the input
687             MessageDigest hash = MessageDigest.getInstance(digest);
688             hash.update(input);
689 
690             //proceed ....
691             byte[] digest = hash.digest();
692 
693             return new String(Hex.encode(digest));
694         }
695         catch (NoSuchAlgorithmException e)
696         {
697             System.err.println("No such algorithm");
698             e.printStackTrace();
699         }
700 
701         return null;
702     }
