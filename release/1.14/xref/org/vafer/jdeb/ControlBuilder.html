50  /**
51   * Builds the control archive of the Debian package.
52   */
53  class ControlBuilder {
54  
55      /** The name of the package maintainer scripts */
56      private static final Set<String> MAINTAINER_SCRIPTS = new HashSet<>(Arrays.asList("preinst", "postinst", "prerm", "postrm", "config"));
57  
58      /** The name of the other control files subject to token substitution */
59      private static final Set<String> CONFIGURATION_FILENAMES = new HashSet<>(Arrays.asList("conffiles", "templates", "triggers", "copyright"));
60  
61      private Console console;
62      private VariableResolver resolver;
63      private final String openReplaceToken;
64      private final String closeReplaceToken;
65      private final Long outputTimestampMs;
66  
67      ControlBuilder(Console console, VariableResolver resolver, String openReplaceToken, String closeReplaceToken, Long outputTimestampMs) {
68          this.console = console;
69          this.resolver = resolver;
70          this.openReplaceToken = openReplaceToken;
71          this.closeReplaceToken = closeReplaceToken;
72          this.outputTimestampMs = outputTimestampMs;
73      }

75      /**
76       * Build control archive of the deb
77       *
78       * @param packageControlFile the package control file
79       * @param controlFiles the other control information files (maintainer scripts, etc)
80       * @param conffiles    the configuration files
81       * @param checksums the md5 checksums of the files in the data archive
82       * @param output
83       * @return
84       * @throws java.io.FileNotFoundException
85       * @throws java.io.IOException
86       * @throws java.text.ParseException
87       */
88      void buildControl(BinaryPackageControlFile packageControlFile, File[] controlFiles, List<String> conffiles, StringBuilder checksums, File output, Charset encoding) throws IOException, ParseException {
89  
90          if (packageControlFile == null) {
91              throw new FileNotFoundException("No 'control' file found in " + controlFiles.toString());
92          }
93  
94          final File dir = output.getParentFile();
95          if (dir != null && (!dir.exists() || !dir.isDirectory())) {
96              throw new IOException("Cannot write control file at '" + output.getAbsolutePath() + "'");
97          }
98  
99          final TarArchiveOutputStream outputStream = new TarArchiveOutputStream(new GZIPOutputStream(new FileOutputStream(output)));
100         outputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
101 
102         boolean foundConffiles = false;
103 
104         // sort files to have consistent/reproducible builds
105         Arrays.sort(controlFiles, Comparator.comparing(File::toString));
106 
107         // create the final package control file out of the "control" file, copy all other files, ignore the directories
108         for (File file : controlFiles) {
109             if (file.isDirectory()) {
110                 // warn about the misplaced directory, except for directories ignored by default (.svn, cvs, etc)
111                 if (!isDefaultExcludes(file)) {
112                     console.warn("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
113                 }
114                 continue;
115             }
116 
117             if ("conffiles".equals(file.getName())) {
118                 foundConffiles = true;
119             }
120 
121             if ("control".equals(file.getName())) {
122                 continue;
123             }
124 
125             if (CONFIGURATION_FILENAMES.contains(file.getName()) || MAINTAINER_SCRIPTS.contains(file.getName())) {
126 
127                 FilteredFile configurationFile = new FilteredFile(new FileInputStream(file), resolver, encoding, openReplaceToken, closeReplaceToken);
128                 addControlEntry(file.getName(), configurationFile.toString(), outputStream);
129 
130             } else {
131 
132                 // initialize the information stream to guess the type of the file
133                 InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
134                 Utils.copy(infoStream, NullOutputStream.INSTANCE);
135                 infoStream.close();
136 
137                 // fix line endings for shell scripts
138                 InputStream in = new FileInputStream(file);
139                 if (infoStream.isShell() && !infoStream.hasUnixLineEndings()) {
140                     byte[] buf = Utils.toUnixLineEndings(in, encoding);
141                     in = new ByteArrayInputStream(buf);
142                 }
143 
144                 addControlEntry(file.getName(), IOUtils.toString(in, encoding), outputStream);
145 
146                 in.close();
147             }
148         }
149 
150         if (foundConffiles) {
151             console.info("Found file 'conffiles' in the control directory. Skipping conffiles generation.");
152         } else if ((conffiles != null) && (conffiles.size() > 0)) {
153             addControlEntry("conffiles", createPackageConffilesFile(conffiles), outputStream);
154         } else {
155             console.info("Skipping 'conffiles' generation. No entries defined in maven/pom or ant/build.xml.");
156         }
157 
158         addControlEntry("control", packageControlFile.toString(), outputStream);
159         addControlEntry("md5sums", checksums.toString(), outputStream);
160 
161         outputStream.close();
162     }
163 
164     private String createPackageConffilesFile(final List<String> conffiles) {
165         StringBuilder content = new StringBuilder();
166 
167         if (conffiles != null && !conffiles.isEmpty()) {
168             for (String nextFileName : conffiles) {
169                 content.append(nextFileName).append("\n");
170             }
171         }
172 
173         return content.toString();
174     }

181     /**
182      * Creates a package control file from the specified file and adds the
183      * <tt>Date</tt>, <tt>Distribution</tt> and <tt>Urgency</tt> fields if missing.
184      * The <tt>Installed-Size</tt> field is also initialized to the actual size of
185      * the package. The <tt>Maintainer</tt> field is overridden by the <tt>DEBEMAIL</tt>
186      * and <tt>DEBFULLNAME</tt> environment variables if defined.
187      *
188      * @param file       the control file
189      * @param pDataSize  the size of the installed package
190      * @param encoding   the encoding used to read the files
191      */
192     public BinaryPackageControlFile createPackageControlFile(File file, BigInteger pDataSize, Charset encoding, String openToken, String closeToken) throws IOException, ParseException {
193         FilteredFile controlFile = new FilteredFile(new FileInputStream(file), resolver, encoding, openToken, closeToken);
194         BinaryPackageControlFile packageControlFile = new BinaryPackageControlFile(controlFile.toString());
195 
196         if (packageControlFile.get("Distribution") == null) {
197             packageControlFile.set("Distribution", "unknown");
198         }
199 
200         if (packageControlFile.get("Urgency") == null) {
201             packageControlFile.set("Urgency", "low");
202         }
203 
204         packageControlFile.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
205 
206         // override the Version if the DEBVERSION environment variable is defined
207         final String debVersion = System.getenv("DEBVERSION");
208         if (debVersion != null) {
209             packageControlFile.set("Version", debVersion);
210             console.debug("Using version'" + debVersion + "' from the environment variables.");
211         }
212 
213 
214         // override the Maintainer field if the DEBFULLNAME and DEBEMAIL environment variables are defined
215         final String debFullName = System.getenv("DEBFULLNAME");
216         final String debEmail = System.getenv("DEBEMAIL");
217 
218         if (debFullName != null && debEmail != null) {
219             final String maintainer = debFullName + " <" + debEmail + ">";
220             packageControlFile.set("Maintainer", maintainer);
221             console.debug("Using maintainer '" + maintainer + "' from the environment variables.");
222         }
223 
224         return packageControlFile;
225     }

228     private void addControlEntry(final String pName, final String pContent, final TarArchiveOutputStream pOutput) throws IOException {
229 
230         console.info("Adding control: " + pName);
231 
232         final byte[] data = pContent.getBytes(UTF_8);
233 
234         final TarArchiveEntry entry = new TarArchiveEntry("./" + pName, true);
235         entry.setSize(data.length);
236         entry.setNames("root", "root");
237         if (outputTimestampMs != null) {
238             entry.setModTime(outputTimestampMs);
239         }
240 
241         if (MAINTAINER_SCRIPTS.contains(pName)) {
242             entry.setMode(PermMapper.toMode("755"));
243         } else {
244             entry.setMode(PermMapper.toMode("644"));
245         }
246 
247         pOutput.putArchiveEntry(entry);
248         pOutput.write(data);
249         pOutput.closeArchiveEntry();
250     }

252     /**
253      * Tells if the specified directory is ignored by default (.svn, cvs, etc)
254      *
255      * @param directory
256      */
257     private boolean isDefaultExcludes(File directory) {
258         for (String pattern : DirectoryScanner.getDefaultExcludes()) {
259             if (DirectoryScanner.match(pattern, directory.getAbsolutePath().replace("\\", "/"))) {
260                 return true;
261             }
262         }
263 
264         return false;
265     }
