017    package org.vafer.jdeb.utils;
018    
019    import java.io.IOException;
020    import java.io.InputStream;
021    
022    public final class InformationInputStream extends InputStream {
023    
024        private final InputStream inputStream;
025    
026        private final char[] shell_utf8 = { 0x23, 0x21 };
027        private final char[] shell_utf16be = { 0x00, 0x23, 0x00, 0x21 };
028        private final char[] shell_utf16le = { 0x23, 0x00, 0x21, 0x00 };
029        private final char[] bom1 = { 0xEF, 0xBB, 0xBF };
030        private final char[] bom2 = { 0xFF, 0xFE };
031        private final char[] bom3 = { 0xFE, 0xFF };
032    
033        private int shell_utf8_i;
034        private int shell_utf16be_i;
035        private int shell_utf16le_i;
036        private int bom1_i;
037        private int bom2_i;
038        private int bom3_i;
039    
040        private long i;
041        private long ascii;
042        private long nonascii;
043        private long cr;
044        private long lf;
045        private long zero;
046    
047        private enum BOM { NONE, BOM1, BOM2, BOM3 };
048        private enum SHELL { NONE, UTF8, UTF16BE, UTF16LE };
049    
050        private BOM bom = BOM.NONE;
051        private SHELL shell = SHELL.NONE;
052    
053        public InformationInputStream(InputStream inputStream) {
054            this.inputStream = inputStream;
055        }
056    
057        public boolean hasBom() {
058            return bom != BOM.NONE;
059        }
060    
061        public boolean isShell() {
062            return shell != SHELL.NONE;
063        }
064        public boolean hasUnixLineEndings() {
065            return cr == 0;
066        }
067    
068        private void add(int c) {
069            // System.out.println(i + ": " + Integer.toHexString(c));
070            if (i < 10) {
071                // FIXME refactor to make this DRY
072                if (shell == SHELL.NONE) {
073                    if (shell_utf8_i < shell_utf8.length) {
074                        if (c == shell_utf8[shell_utf8_i]) {
075                            shell_utf8_i++;
076                        } else {
077                            shell_utf8_i = 0;
078                        }
079                    } else {
080                        // System.out.println("utf8 shell at " + (i-shell_utf8.length));
081                        shell = SHELL.UTF8;
082                    }
083    
084                    if (shell_utf16be_i < shell_utf16be.length) {
085                        if (c == shell_utf16be[shell_utf16be_i]) {
086                            shell_utf16be_i++;
087                        } else {
088                            shell_utf16be_i = 0;
089                        }
090                    } else {
091                        // System.out.println("utf16be shell at " + (i-shell_utf16be.length));
092                        shell = SHELL.UTF16BE;
093                    }
094    
095                    if (shell_utf16le_i < shell_utf16le.length) {
096                        if (c == shell_utf16le[shell_utf16le_i]) {
097                            shell_utf16le_i++;
098                        } else {
099                            shell_utf16le_i = 0;
100                        }
101                    } else {
102                        // System.out.println("utf16le shell at " + (i-shell_utf16le.length));
103                        shell = SHELL.UTF16LE;
104                    }
105                }
106                if (bom == BOM.NONE) {
107                    if (bom1_i < bom1.length) {
108                        if (c == bom1[bom1_i] && bom1_i == i) {
109                            bom1_i++;
110                        } else {
111                            bom1_i = 0;
112                        }
113                    } else {
114                        // System.out.println("bom1 at " + (i-bom1.length));
115                        bom = BOM.BOM1;
116                    }
117    
118                    if (bom2_i < bom2.length) {
119                        if (c == bom2[bom2_i] && bom2_i == i) {
120                            bom2_i++;
121                        } else {
122                            bom2_i = 0;
123                        }
124                    } else {
125                        // System.out.println("bom2 at " + (i-bom2.length));
126                        bom = BOM.BOM2;
127                    }
128    
129                    if (bom3_i < bom3.length) {
130                        if (c == bom3[bom3_i] && bom3_i == i) {
131                            bom3_i++;
132                        } else {
133                            bom3_i = 0;
134                        }
135                    } else {
136                        // System.out.println("bom3 at " + (i-bom3.length));
137                        bom = BOM.BOM3;
138                    }
139    
140                }
141            }
142    
143            i++;
144    
145            if (c == '\n') {
146                lf++;
147                return;
148            }
149            if (c == '\r') {
150                cr++;
151                return;
152            }
153            if (c >= ' ' && c <= '~') {
154                ascii++;
155                return;
156            }
157            if (c == 0) {
158                zero++;
159                return;
160            }
161            nonascii++;
162        }
163    
164        public int available() throws IOException {
165            return inputStream.available();
166        }
167    
168        public void close() throws IOException {
169            inputStream.close();
170        }
171    
172        public void mark(int readlimit) {
173            inputStream.mark(readlimit);
174        }
175    
176        public boolean markSupported() {
177            return inputStream.markSupported();
178        }
179    
180        public int read() throws IOException {
181            int ret = inputStream.read();
182            if (ret != -1) {
183                add(ret & 0xFF);
184            }
185            return ret;
186        }
187    
188        public int read(byte[] b, int off, int len) throws IOException {
189            int ret = inputStream.read(b, off, len);
190            for(int i = 0; i<ret; i++) {
191                add(b[off+i] & 0xFF);
192            }
193            return ret;
194        }
195    
196        public int read(byte[] b) throws IOException {
197            return read(b, 0, b.length);
198        }
199    
200        public void reset() throws IOException {
201            inputStream.reset();
202        }
203    
204        public long skip(long n) throws IOException {
205            return inputStream.skip(n);
206        }
207    
208        public String toString() {
209            StringBuilder sb = new StringBuilder();
210            sb.append("{");
211            sb.append("total=").append(i);
212            sb.append(",noascii=").append(nonascii);
213            sb.append(",ascii=").append(ascii);
214            sb.append(",cr=").append(cr);
215            sb.append(",lf=").append(lf);
216            sb.append(",zero=").append(zero);
217            sb.append("}");
218            return sb.toString();
219        }
220    }
