016package org.vafer.jdeb.ant;
017
018import java.io.File;
019import java.io.FileNotFoundException;
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Iterator;
024
025import org.apache.tools.ant.types.PatternSet;
026import org.vafer.jdeb.DataConsumer;
027import org.vafer.jdeb.DataProducer;
028import org.vafer.jdeb.producers.DataProducerArchive;
029import org.vafer.jdeb.producers.DataProducerDirectory;
030import org.vafer.jdeb.producers.DataProducerFile;
031
032/**
033 * Ant "data" element acting as a factory for DataProducers.
034 * So far Archive and Directory producers are supported.
035 * Both support the usual ant pattern set matching.
036 */
037public final class Data extends PatternSet implements DataProducer {
038
039    private final Collection<Mapper> mapperWrapper = new ArrayList<Mapper>();
040
041    private File src;
042
043    private String type;
044
045    private Boolean conffile;
046
047    private String destinationName;
048
049    public void setSrc(File src) {
050        this.src = src;
051    }
052
053    public String getType() {
054        return type;
055    }
056
057    public void setType(String type) {
058        this.type = type;
059    }
060
061    public void setConffile(Boolean conffile) {
062        this.conffile = conffile;
063    }
064    
065    public Boolean getConffile() {
066        return this.conffile;
067    }
068
069    public void setDst(String destinationName) {
070        this.destinationName = destinationName;
071    }
072
073    public void addMapper(Mapper mapper) {
074        mapperWrapper.add(mapper);
075    }
076
077    public void produce( final DataConsumer pReceiver ) throws IOException {
078
079        if (src == null || !src.exists()) {
080            throw new FileNotFoundException("Data source not found : " + src);
081        }
082
083        org.vafer.jdeb.mapping.Mapper[] mappers = new org.vafer.jdeb.mapping.Mapper[mapperWrapper.size()];
084        final Iterator<Mapper> it = mapperWrapper.iterator();
085        for (int i = 0; i < mappers.length; i++) {
086            mappers[i] = it.next().createMapper();
087        }
088
089        if ("file".equalsIgnoreCase(type)) {
090            new DataProducerFile(
091                src,
092                destinationName,
093                getIncludePatterns(getProject()),
094                getExcludePatterns(getProject()),
095                mappers
096            ).produce(pReceiver);
097
098        } else if ("archive".equalsIgnoreCase(type)) {
099            new DataProducerArchive(
100                src,
101                getIncludePatterns(getProject()),
102                getExcludePatterns(getProject()),
103                mappers
104            ).produce(pReceiver);
105
106        } else if ("directory".equalsIgnoreCase(type)) {
107            new DataProducerDirectory(
108                src,
109                getIncludePatterns(getProject()),
110                getExcludePatterns(getProject()),
111                mappers
112            ).produce(pReceiver);
113        }
114    }
115}
