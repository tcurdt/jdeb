17  package org.vafer.jdeb.signing;
18  
19  import java.io.ByteArrayInputStream;
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.io.InputStreamReader;
23  import java.io.OutputStream;
24  import java.nio.charset.Charset;
25  import java.security.GeneralSecurityException;
26  import java.util.Iterator;
27  
28  import org.apache.commons.io.LineIterator;
29  import org.bouncycastle.bcpg.ArmoredOutputStream;
30  import org.bouncycastle.bcpg.BCPGOutputStream;
31  import org.bouncycastle.openpgp.PGPException;
32  import org.bouncycastle.openpgp.PGPPrivateKey;
33  import org.bouncycastle.openpgp.PGPSecretKey;
34  import org.bouncycastle.openpgp.PGPSecretKeyRing;
35  import org.bouncycastle.openpgp.PGPSecretKeyRingCollection;
36  import org.bouncycastle.openpgp.PGPSignature;
37  import org.bouncycastle.openpgp.PGPSignatureGenerator;
38  import org.bouncycastle.openpgp.PGPUtil;
39  import org.bouncycastle.openpgp.operator.bc.BcPBESecretKeyDecryptorBuilder;
40  import org.bouncycastle.openpgp.operator.bc.BcPGPContentSignerBuilder;
41  import org.bouncycastle.openpgp.operator.bc.BcPGPDigestCalculatorProvider;
42  
43  /**
44   * Signing with OpenPGP.
45   */
46  public class PGPSigner {
47  
48      private static final byte[] EOL = "\n".getBytes(Charset.forName("UTF-8"));
49  
50      private PGPSecretKey secretKey;
51      private PGPPrivateKey privateKey;
52  
53      public PGPSigner(InputStream keyring, String keyId, String passphrase) throws IOException, PGPException {
54          secretKey = getSecretKey(keyring, keyId);
55          if(secretKey == null)
56          {
57              throw new PGPException(String.format("Specified key %s does not exist in key ring %s", keyId, keyring));
58          }
59          privateKey = secretKey.extractPrivateKey(new BcPBESecretKeyDecryptorBuilder(new BcPGPDigestCalculatorProvider()).build(passphrase.toCharArray()));
60      }
61  
62      /**
63       * Creates a clear sign signature over the input data. (Not detached)
64       *
65       * @param input      the content to be signed
66       * @param output     the output destination of the signature
67       */
68      public void clearSign(String input, OutputStream output) throws IOException, PGPException, GeneralSecurityException {
69          clearSign(new ByteArrayInputStream(input.getBytes("UTF-8")), output);
70      }
71  
72      /**
73       * Creates a clear sign signature over the input data. (Not detached)
74       *
75       * @param input      the content to be signed
76       * @param output     the output destination of the signature
77       */
78      public void clearSign(InputStream input, OutputStream output) throws IOException, PGPException, GeneralSecurityException {
79          int digest = PGPUtil.SHA1;
80          
81          PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(privateKey.getPublicKeyPacket().getAlgorithm(), digest));
82          signatureGenerator.init(PGPSignature.CANONICAL_TEXT_DOCUMENT, privateKey);
83          
84          ArmoredOutputStream armoredOutput = new ArmoredOutputStream(output);
85          armoredOutput.beginClearText(digest);
86          
87          LineIterator iterator = new LineIterator(new InputStreamReader(input));
88          
89          while (iterator.hasNext()) {
90              String line = iterator.nextLine();
91              
92              // trailing spaces must be removed for signature calculation (see http://tools.ietf.org/html/rfc4880#section-7.1)
93              byte[] data = trim(line).getBytes("UTF-8");
94              
95              armoredOutput.write(data);
96              armoredOutput.write(EOL);
97              
98              signatureGenerator.update(data);
99              if (iterator.hasNext()) {
100                 signatureGenerator.update(EOL);
101             }
102         }
103 
104         armoredOutput.endClearText();
105         
106         PGPSignature signature = signatureGenerator.generate();
107         signature.encode(new BCPGOutputStream(armoredOutput));
108         
109         armoredOutput.close();
110     }
111     
112     /**
113      * Returns the secret key.
114      */
115     public PGPSecretKey getSecretKey()
116     {
117         return secretKey;
118     }
119 
120     /**
121      * Returns the private key.
122      */
123     public PGPPrivateKey getPrivateKey()
124     {
125         return privateKey;
126     }
127 
128     /**
129      * Returns the secret key matching the specified identifier.
130      * 
131      * @param input the input stream containing the keyring collection
132      * @param keyId the 4 bytes identifier of the key
133      */
134     private PGPSecretKey getSecretKey(InputStream input, String keyId) throws IOException, PGPException {
135         PGPSecretKeyRingCollection keyrings = new PGPSecretKeyRingCollection(PGPUtil.getDecoderStream(input));
136         
137         Iterator rIt = keyrings.getKeyRings();
138 
139         while (rIt.hasNext()) {
140             PGPSecretKeyRing kRing = (PGPSecretKeyRing) rIt.next();
141             Iterator kIt = kRing.getSecretKeys();
142 
143             while (kIt.hasNext()) {
144                 PGPSecretKey key = (PGPSecretKey) kIt.next();
145                 
146                 if (key.isSigningKey() && String.format("%08x", key.getKeyID() & 0xFFFFFFFFL).equals(keyId.toLowerCase())) {
147                     return key;
148                 }
149             }
150         }
151 
152         return null;
153     }
154 
155     /**
156      * Trim the trailing spaces.
157      * 
158      * @param line
159      */
160     private String trim(String line) {
161         char[] chars = line.toCharArray();
162         int len = chars.length;
163 
164         while (len > 0) {
165             if (!Character.isWhitespace(chars[len - 1])) {
166                 break;
167             }
168             len--;
169         }
170         
171         return line.substring(0, len);
172     }
173 }
